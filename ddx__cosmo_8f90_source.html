41subroutine ddcosmo(params, constants, workspace, state, phi_cav, &
42 & psi, tol, esolv, force)
43 type(ddx_params_type), intent(in) :: params
44 type(ddx_constants_type), intent(in) :: constants
45 type(ddx_workspace_type), intent(inout) :: workspace
46 type(ddx_state_type), intent(inout) :: state
47 real(dp), intent(in) :: phi_cav(constants % ncav), &
48 & psi(constants % nbasis, params % nsph), tol
49 real(dp), intent(out) :: esolv, force(3, params % nsph)
50 real(dp), external :: ddot
51
52 call ddcosmo_setup(params, constants, workspace, state, phi_cav, psi)
53 call ddcosmo_guess(params, constants, workspace, state)
54 call ddcosmo_solve(params, constants, workspace, state, tol)
55
56 ! Solvation energy is computed
57 esolv = pt5*ddot(constants % n, state % xs, 1, psi, 1)
58
59 ! Get forces if needed
60 if (params % force .eq. 1) then
61 ! solve the adjoint
62 call ddcosmo_guess_adjoint(params, constants, workspace, state)
63 call ddcosmo_solve_adjoint(params, constants, workspace, state, tol)
64
65 ! evaluate the solvent unspecific contribution analytical derivatives
66 force = zero
67 call ddcosmo_solvation_force_terms(params, constants, workspace, &
68 & state, force)
69 end if

205subroutine ddcosmo_solvation_force_terms(params, constants, workspace, &
206 & state, force)
207 implicit none
208 type(ddx_params_type), intent(in) :: params
209 type(ddx_constants_type), intent(in) :: constants
210 type(ddx_workspace_type), intent(inout) :: workspace
211 type(ddx_state_type), intent(inout) :: state
212 real(dp), intent(inout) :: force(3, params % nsph)
213 ! local variables
214 real(dp), external :: ddot
215 integer :: icav, isph, igrid
216
217 ! Get values of S on the grid
218 call ddeval_grid_work(constants % nbasis, params % ngrid, params % nsph, &
219 & constants % vgrid, constants % vgrid_nbasis, one, state % s, zero, &
220 & state % sgrid)
221 ! Get the values of phi on the grid
222 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
223 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
224 & state % phi_grid)
225
226 force = zero
227 do isph = 1, params % nsph
228 call contract_grad_l(params, constants, isph, state % xs, &
229 & state % sgrid, workspace % tmp_vylm(:, 1), &
230 & workspace % tmp_vdylm(:, :, 1), workspace % tmp_vplm(:, 1), &
231 & workspace % tmp_vcos(:, 1), workspace % tmp_vsin(:, 1), &
232 & force(:, isph))
233 call contract_grad_u(params, constants, isph, state % sgrid, &
234 & state % phi_grid, force(:, isph))
235 end do
236 force = - pt5 * force
237
238 ! assemble the intermediate zeta: S weighted by U evaluated on the
239 ! exposed grid points. This is not required here, but it is used
240 ! in later steps.
241 icav = 0
242 do isph = 1, params % nsph
243 do igrid = 1, params % ngrid
244 if (constants % ui(igrid, isph) .ne. zero) then
245 icav = icav + 1
246 state % zeta(icav) = constants % wgrid(igrid) * &
247 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
248 & constants % vgrid(1, igrid), 1, state % s(1, isph), 1)
249 end if
250 end do
251 end do
252
253end subroutine ddcosmo_solvation_force_terms

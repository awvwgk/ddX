155subroutine ddcosmo_solve_worker(params, constants, workspace, phi_cav, &
156 & xs, xs_niter, xs_rel_diff, xs_time, tol, phi_grid, &
157 & phi, info)
158 !! Inputs
159 type(ddx_params_type), intent(in) :: params
160 type(ddx_constants_type), intent(in) :: constants
161 real(dp), intent(in) :: phi_cav(constants % ncav), tol
162 !! Input+output
163 real(dp), intent(inout) :: xs(constants % nbasis, params % nsph)
164 integer, intent(inout) :: xs_niter
165 !! Temporary buffers
166 type(ddx_workspace_type), intent(inout) :: workspace
167 !! Outputs
168 real(dp), intent(out) :: xs_rel_diff(xs_niter), xs_time
169 real(dp), intent(out) :: phi_grid(params % ngrid, params % nsph)
170 real(dp), intent(out) :: phi(constants % nbasis, params % nsph)
171 integer, intent(out) :: info
172 !! Local variables
173 character(len=255) :: string
174 real(dp) :: start_time, finish_time, r_norm
175 !! The code
176 ! At first check if parameters, constants and workspace are correctly
177 ! initialized
178 if (params % error_flag .ne. 0) then
179 string = "ddcosmo_energy: `params` is in error state"
180 call params % print_func(string)
181 info = 1
182 return
183 end if
184 if (constants % error_flag .ne. 0) then
185 string = "ddcosmo_energy: `constants` is in error state"
186 call params % print_func(string)
187 info = 1
188 return
189 end if
190 if (workspace % error_flag .ne. 0) then
191 string = "ddcosmo_energy: `workspace` is in error state"
192 call params % print_func(string)
193 info = 1
194 return
195 end if
196 ! Unwrap sparsely stored potential at cavity points phi_cav into phi_grid
197 ! and multiply it by characteristic function at cavity points ui
198 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
199 & constants % icav_ia, constants % icav_ja, phi_cav, phi_grid)
200 workspace % tmp_cav = phi_cav * constants % ui_cav
201 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
202 & constants % icav_ia, constants % icav_ja, workspace % tmp_cav, &
203 & workspace % tmp_grid)
204 ! Integrate against spherical harmonics and Lebedev weights to get Phi
205 call ddintegrate(params % nsph, constants % nbasis, &
206 & params % ngrid, constants % vwgrid, &
207 & constants % vgrid_nbasis, workspace % tmp_grid, phi)
208! call ddintegrate_sph_work(constants % nbasis, params % ngrid, &
209! & params % nsph, constants % vwgrid, constants % vgrid_nbasis, &
210! & one, workspace % tmp_grid, zero, phi)
211 ! Set right hand side to -Phi
212 workspace % tmp_rhs = -phi
213 ! Solve ddCOSMO system L X = -Phi with a given initial guess
214 start_time = omp_get_wtime()
215 if (params % itersolver .eq. 1) then
216 call jacobi_diis(params, constants, workspace, tol, &
217 & workspace % tmp_rhs, xs, xs_niter, xs_rel_diff, lx, &
218 & ldm1x, hnorm, info)
219 else
220 call gmresr(params, constants, workspace, tol, &
221 & workspace % tmp_rhs, xs, xs_niter, r_norm, lx, info)
222 end if
223 finish_time = omp_get_wtime()
224 xs_time = finish_time - start_time
225 ! Check if solver did not converge
226 if (info .ne. 0) then
227 string = "ddcosmo_energy: solver for ddCOSMO system did not converge"
228 call params % print_func(string)
229 return
230 end if
231 ! Clear status
232 info = 0
233end subroutine ddcosmo_solve_worker

303subroutine ddcosmo_forces_worker(params, constants, workspace, phi_grid, &
304 & gradphi_cav, psi, s, sgrid, xs, zeta, force, info)
305 !! Inputs
306 type(ddx_params_type), intent(in) :: params
307 type(ddx_constants_type), intent(in) :: constants
308 real(dp), intent(in) :: phi_grid(params % ngrid, params % nsph), &
309 & gradphi_cav(3, constants % ncav), &
310 & psi(constants % nbasis, params % nsph), &
311 & s(constants % nbasis, params % nsph), &
312 & xs(constants % nbasis, params % nsph)
313 !! Temporary buffers
314 type(ddx_workspace_type), intent(inout) :: workspace
315 !! Outputs
316 real(dp), intent(out) :: sgrid(params % ngrid, params % nsph), &
317 & zeta(constants % ncav), &
318 & force(3, params % nsph)
319 integer, intent(out) :: info
320 !! Local variables
321 integer :: isph, icav, igrid, inode, jnode, jsph, jnear
322 real(dp) :: tmp1, tmp2, d(3), dnorm
323 real(dp), external :: ddot, dnrm2
324 character(len=255) :: string
325 !! The code
326 ! At first check if parameters, constants and workspace are correctly
327 ! initialized
328 if (params % error_flag .ne. 0) then
329 string = "ddcosmo_forces: `params` is in error state"
330 call params % print_func(string)
331 info = 1
332 return
333 end if
334 if (constants % error_flag .ne. 0) then
335 string = "ddcosmo_forces: `constants` is in error state"
336 call params % print_func(string)
337 info = 1
338 return
339 end if
340 if (workspace % error_flag .ne. 0) then
341 string = "ddcosmo_forces: `workspace` is in error state"
342 call params % print_func(string)
343 info = 1
344 return
345 end if
346 ! Get values of S on grid
347 call ddeval_grid_work(constants % nbasis, params % ngrid, params % nsph, &
348 & constants % vgrid, constants % vgrid_nbasis, one, s, zero, sgrid)
349 force = zero
350 do isph = 1, params % nsph
351 call contract_grad_l(params, constants, isph, xs, sgrid, &
352 & workspace % tmp_vylm(:, 1), workspace % tmp_vdylm(:, :, 1), &
353 & workspace % tmp_vplm(:, 1), &
354 & workspace % tmp_vcos(:, 1), &
355 & workspace % tmp_vsin(:, 1), force(:, isph))
356 call contract_grad_u(params, constants, isph, sgrid, phi_grid, &
357 & force(:, isph))
358 end do
359 force = - pt5 * force
360
361 icav = 0
362 do isph = 1, params % nsph
363 do igrid = 1, params % ngrid
364 if (constants % ui(igrid, isph) .ne. zero) then
365 icav = icav + 1
366 zeta(icav) = constants % wgrid(igrid) * &
367 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
368 & constants % vgrid(1, igrid), 1, &
369 & s(1, isph), 1)
370 force(:, isph) = force(:, isph) &
371 & - pt5*zeta(icav)*gradphi_cav(:, icav)
372 end if
373 end do
374 end do
375
376 !! Last term where we compute gradients of potential at centers of atoms
377 !! spawned by intermediate zeta.
378 if(params % fmm .eq. 1) then
379 !! This step can be substituted by a proper dgemm if zeta
380 !! intermediate is converted from cavity points to all grid points
381 !! with zero values at internal grid points
382 ! P2M step
383 icav = 0
384 do isph = 1, params % nsph
385 inode = constants % snode(isph)
386 workspace % tmp_node_m(:, inode) = zero
387 do igrid = 1, params % ngrid
388 if(constants % ui(igrid, isph) .eq. zero) cycle
389 icav = icav + 1
390 call fmm_p2m(constants % cgrid(:, igrid), zeta(icav), &
391 & one, params % pm, constants % vscales, one, &
392 & workspace % tmp_node_m(:, inode))
393 end do
394 workspace % tmp_node_m(:, inode) = workspace % tmp_node_m(:, inode) / &
395 & params % rsph(isph)
396 end do
397 ! M2M, M2L and L2L translations
398 call tree_m2m_rotation(params, constants, workspace % tmp_node_m)
399 call tree_m2l_rotation(params, constants, workspace % tmp_node_m, &
400 & workspace % tmp_node_l)
401 call tree_l2l_rotation(params, constants, workspace % tmp_node_l)
402 ! Now compute near-field FMM gradients
403 ! Cycle over all spheres
404 icav = 0
405 workspace % tmp_efld = zero
406 do isph = 1, params % nsph
407 ! Cycle over all external grid points
408 do igrid = 1, params % ngrid
409 if (constants % ui(igrid, isph) .eq. zero) cycle
410 icav = icav + 1
411 ! Cycle over all near-field admissible pairs of spheres,
412 ! including pair (isph, isph) which is a self-interaction
413 inode = constants % snode(isph)
414 do jnear = constants % snear(inode), constants % snear(inode+1)-1
415 ! Near-field interactions are possible only between leaf
416 ! nodes, which must contain only a single input sphere
417 jnode = constants % near(jnear)
418 jsph = constants % order(constants % cluster(1, jnode))
419 d = params % csph(:, isph) + &
420 & constants % cgrid(:, igrid)*params % rsph(isph) - &
421 & params % csph(:, jsph)
422 dnorm = dnrm2(3, d, 1)
423 workspace % tmp_efld(:, jsph) = workspace % tmp_efld(:, jsph) + &
424 & zeta(icav)*d/(dnorm**3)
425 end do
426 end do
427 end do
428 ! Take into account far-field FMM gradients only if pl > 0
429 if (params % pl .gt. 0) then
430 tmp1 = one / sqrt(3d0) / constants % vscales(1)
431 do isph = 1, params % nsph
432 inode = constants % snode(isph)
433 tmp2 = tmp1 / params % rsph(isph)
434 workspace % tmp_efld(3, isph) = workspace % tmp_efld(3, isph) + &
435 & tmp2*workspace % tmp_node_l(3, inode)
436 workspace % tmp_efld(1, isph) = workspace % tmp_efld(1, isph) + &
437 & tmp2*workspace % tmp_node_l(4, inode)
438 workspace % tmp_efld(2, isph) = workspace % tmp_efld(2, isph) + &
439 & tmp2*workspace % tmp_node_l(2, inode)
440 end do
441 end if
442 do isph = 1, params % nsph
443 force(:, isph) = force(:, isph) &
444 & -pt5*workspace % tmp_efld(:, isph)*params % charge(isph)
445 end do
446 ! Naive quadratically scaling implementation
447 else
448 ! This routines actually computes -grad, not grad
449 call efld(constants % ncav, zeta, constants % ccav, &
450 & params % nsph, params % csph, workspace % tmp_efld)
451 do isph = 1, params % nsph
452 force(:, isph) = force(:, isph) &
453 & + pt5*workspace % tmp_efld(:, isph)*params % charge(isph)
454 end do
455 end if
456 ! Clear status
457 info = 0
458end subroutine ddcosmo_forces_worker

460subroutine ddcosmo_geom_forces_worker(params, constants, workspace, phi_grid, &

525 icav = icav + 1

527 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
528 & constants % vgrid(1, igrid), 1, &
529 & s(1, isph), 1)

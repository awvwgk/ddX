241subroutine constants_init(params, constants)
242 use complex_bessel
243 !! Inputs
244 type(ddx_params_type), intent(in) :: params
245 !! Outputs
246 type(ddx_constants_type), intent(out) :: constants
247 !! Local variables
248 integer :: i, alloc_size, l, indl, igrid, isph, l0, &
249 & NZ, ierr, info, tmp_pmax
250 real(dp) :: rho, ctheta, stheta, cphi, sphi, termi, termk, &
251 & s1, s2
252 real(dp), allocatable :: vplm(:), vcos(:), vsin(:), vylm(:), SK_rijn(:), &
253 & DK_rijn(:)
254 complex(dp), allocatable :: bessel_work(:)
255 complex(dp) :: z
256 !! The code
257 ! Clean error state of constants to proceed with geometry
258 constants % error_flag = 0
259 constants % error_message = ""
260 ! Check if params are OK
261 if (params % error_flag .ne. 0) then
262 constants % error_flag = 1
263 constants % error_message = "constants_init: `params` is in " // &
264 & "error state"
265 return
266 end if
267 ! activate inner iterations diagonal in mvp for debugging purposes only.
268 ! could be useful for different linear solvers.
269 constants % dodiag = .false.
270 ! Maximal number of modeling spherical harmonics
271 constants % nbasis = (params % lmax+1) ** 2
272 ! Maximal number of modeling degrees of freedom
273 constants % n = params % nsph * constants % nbasis
274 ! Calculate dmax, vgrid_dmax, m2p_lmax, m2p_nbasis and grad_nbasis
275 if (params % fmm .eq. 0) then
276 if (params % force .eq. 1) then
277 constants % dmax = params % lmax + 1
278 constants % vgrid_dmax = params % lmax + 1
279 else
280 constants % dmax = params % lmax
281 constants % vgrid_dmax = params % lmax
282 end if
283 ! Other constants are not referenced if fmm=0
284 constants % m2p_lmax = -1
285 constants % m2p_nbasis = -1
286 constants % grad_nbasis = -1
287 else
288 ! If forces are required then we need the M2P of a degree lmax+1 for
289 ! the near-field analytical gradients
290 if (params % force .eq. 1) then
291 constants % dmax = max(params % pm+params % pl+1, &
292 & params % lmax+1)
293 constants % m2p_lmax = params % lmax + 1
294 constants % grad_nbasis = (params % lmax+2) ** 2
295 constants % vgrid_dmax = max(params % pl, params % lmax) + 1
296 else
297 constants % dmax = max(params % pm+params % pl, &
298 & params % lmax)
299 constants % m2p_lmax = params % lmax
300 constants % grad_nbasis = -1
301 constants % vgrid_dmax = max(params % pl, params % lmax)
302 end if
303 constants % m2p_nbasis = (constants % m2p_lmax+1) ** 2
304 end if
305 ! Compute sizes of vgrid, vfact and vscales
306 constants % vgrid_nbasis = (constants % vgrid_dmax+1) ** 2
307 constants % nfact = 2*constants % dmax+1
308 constants % nscales = (constants % dmax+1) ** 2
309 ! Allocate space for scaling factors of spherical harmonics
310 allocate(constants % vscales(constants % nscales), stat=info)
311 if (info .ne. 0) then
312 constants % error_flag = 1
313 constants % error_message = "constants_init: `vscales` allocation " &
314 & // "failed"
315 return
316 end if
317 allocate(constants % v4pi2lp1(constants % dmax+1), stat=info)
318 if (info .ne. 0) then
319 constants % error_flag = 1
320 constants % error_message = "constants_init: `v4pi2lp1` allocation " &
321 & // "failed"
322 return
323 end if
324 allocate(constants % vscales_rel(constants % nscales), stat=info)
325 if (info .ne. 0) then
326 constants % error_flag = 1
327 constants % error_message = "constants_init: `vscales_rel` " // &
328 & "allocation failed"
329 return
330 end if
331 ! Compute scaling factors of spherical harmonics
332 call ylmscale(constants % dmax, constants % vscales, &
333 & constants % v4pi2lp1, constants % vscales_rel)
334 ! Allocate square roots of factorials
335 allocate(constants % vfact(constants % nfact), stat=info)
336 if (info .ne. 0) then
337 constants % error_flag = 1
338 constants % error_message = "constants_init: `vfact` allocation failed"
339 return
340 end if
341 ! Compute square roots of factorials
342 constants % vfact(1) = 1
343 do i = 2, constants % nfact
344 constants % vfact(i) = constants % vfact(i-1) * sqrt(dble(i-1))
345 end do
346 ! Allocate square roots of combinatorial numbers C_n^k and M2L OZ
347 ! translation coefficients
348 if (params % fmm .eq. 1 .or. &
349 & (params % model .eq. 3 .and. params % force .eq. 1)) then
350 alloc_size = 2*constants % dmax + 1
351 alloc_size = alloc_size * (constants % dmax+1)
352 ! Allocate C_n^k
353 allocate(constants % vcnk(alloc_size), stat=info)
354 if (info .ne. 0) then
355 constants % error_flag = 1
356 constants % error_message = "constants_init: `vcnk` allocation " &
357 & // "failed"
358 return
359 end if
360 ! Allocate M2L OZ coefficients
361 allocate(constants % m2l_ztranslate_coef(&
362 & (params % pm+1), (params % pl+1), (params % pl+1)), &
363 & stat=info)
364 if (info .ne. 0) then
365 constants % error_flag = 1
366 constants % error_message = "constants_init: " // &
367 & "`m2l_ztranslate_coef` allocation failed"
368 return
369 end if
370 ! Allocate adjoint M2L OZ coefficients
371 allocate(constants % m2l_ztranslate_adj_coef(&
372 & (params % pl+1), (params % pl+1), (params % pm+1)), &
373 & stat=info)
374 if (info .ne. 0) then
375 constants % error_flag = 1
376 constants % error_message = "constants_init: " // &
377 & "`m2l_ztranslate_adj_coef` allocation failed"
378 return
379 end if
380 ! Compute combinatorial numbers C_n^k and M2L OZ translate coefficients
381 call fmm_constants(constants % dmax, params % pm, &
382 & params % pl, constants % vcnk, &
383 & constants % m2l_ztranslate_coef, &
384 & constants % m2l_ztranslate_adj_coef)
385 end if
386 ! Allocate space for Lebedev grid coordinates and weights
387 allocate(constants % cgrid(3, params % ngrid), &
388 & constants % wgrid(params % ngrid), stat=info)
389 if (info .ne. 0) then
390 constants % error_flag = 1
391 constants % error_message = "constants_init: `cgrid` and `wgrid` " // &
392 & "allocations failed"
393 return
394 end if
395 ! Get weights and coordinates of Lebedev grid points
396 call llgrid(params % ngrid, constants % wgrid, constants % cgrid)
397 ! Allocate space for values of non-weighted and weighted spherical
398 ! harmonics and L2P at Lebedev grid points
399 allocate(constants % vgrid(constants % vgrid_nbasis, params % ngrid), &
400 & constants % vwgrid(constants % vgrid_nbasis, params % ngrid), &
401 & stat=info)
402 if (info .ne. 0) then
403 constants % error_flag = 1
404 constants % error_message = "constants_init: `vgrid`, `wgrid` and" // &
405 & " allocations failed"
406 return
407 end if
408 allocate(vplm(constants % vgrid_nbasis), vcos(constants % vgrid_dmax+1), &
409 & vsin(constants % vgrid_dmax+1), stat=info)
410 if (info .ne. 0) then
411 constants % error_flag = 1
412 constants % error_message = "constants_init: `vplm`, `vcos` and " // &
413 & "`vsin` allocations failed"
414 return
415 end if
416 ! Compute non-weighted and weighted spherical harmonics and the single
417 ! layer operator at Lebedev grid points
418 do igrid = 1, params % ngrid
419 call ylmbas(constants % cgrid(:, igrid), rho, ctheta, stheta, cphi, &
420 & sphi, constants % vgrid_dmax, constants % vscales, &
421 & constants % vgrid(:, igrid), vplm, vcos, vsin)
422 constants % vwgrid(:, igrid) = constants % wgrid(igrid) * &
423 & constants % vgrid(:, igrid)
424 end do
425 if (params % fmm .eq. 1) then
426 allocate(constants % vgrid2( &
427 & constants % vgrid_nbasis, params % ngrid), &
428 & stat=info)
429 if (info .ne. 0) then
430 constants % error_flag = 1
431 constants % error_message = "constants_init: `vgrid2` " // &
432 & "allocation failed"
433 return
434 end if
435 do igrid = 1, params % ngrid
436 do l = 0, constants % vgrid_dmax
437 indl = l*l + l + 1
438 constants % vgrid2(indl-l:indl+l, igrid) = &
439 & constants % vgrid(indl-l:indl+l, igrid) / &
440 & constants % vscales(indl)**2
441 end do
442 end do
443 end if
444 ! Generate geometry-related constants (required by the LPB code)
445 call constants_geometry_init(params, constants)
446 if (constants % error_flag .ne. 0) return
447 ! Precompute LPB-related constants
448 if (params % model .eq. 3) then
449 constants % lmax0 = params % lmax
450 constants % nbasis0 = constants % nbasis
451 allocate(vylm(constants % vgrid_nbasis), &
452 & sk_rijn(0:constants % lmax0), dk_rijn(0:constants % lmax0), &
453 & bessel_work(constants % dmax+2), stat=info)
454 if (info .ne. 0) then
455 constants % error_flag = 1
456 constants % error_message = "constants_init: `vylm`, `SK_rijn` and " // &
457 & "`DK_rijn` allocations failed"
458 return
459 end if
460 allocate(constants % SI_ri(0:constants % dmax+1, params % nsph))
461 allocate(constants % DI_ri(0:constants % dmax+1, params % nsph))
462 allocate(constants % SK_ri(0:params % lmax+1, params % nsph))
463 allocate(constants % DK_ri(0:params % lmax+1, params % nsph))
464 allocate(constants % Pchi(constants % nbasis, constants % nbasis0, &
465 & params % nsph))
466 allocate(constants % C_ik(0:params % lmax, params % nsph))
467 allocate(constants % termimat(0:params % lmax, params % nsph))
468 sk_rijn = zero
469 dk_rijn = zero
470 do isph = 1, params % nsph
471 ! We compute Bessel functions of degrees 0..lmax+1 because the
472 ! largest degree is required for forces
473 call modified_spherical_bessel_first_kind(constants % dmax+1, &
474 & params % rsph(isph)*params % kappa,&
475 & constants % SI_ri(:, isph), constants % DI_ri(:, isph), &
476 & bessel_work)
477 call modified_spherical_bessel_second_kind(params % lmax+1, &
478 & params % rsph(isph)*params % kappa, &
479 & constants % SK_ri(:, isph), constants % DK_ri(:, isph), &
480 & bessel_work)
481 ! Compute matrix PU_i^e(x_in)
482 ! Previous implementation in update_rhs. Made it in ddinit, so as to use
483 ! it in Forces as well.
484 call mkpmat(params, constants, isph, constants % Pchi(:, :, isph))
485 ! Compute i'_l(r_i)/i_l(r_i)
486 do l = 0, params % lmax
487 constants % termimat(l, isph) = constants % DI_ri(l, isph) / &
488 & constants % SI_ri(l, isph) * params % kappa
489 end do
490 ! Compute (i'_l0/i_l0 - k'_l0/k_l0)^(-1) is computed in Eq.(97)
491 do l0 = 0, constants % lmax0
492 termi = constants % DI_ri(l0, isph) / &
493 & constants % SI_ri(l0, isph) * params % kappa
494 termk = constants % DK_ri(l0, isph)/ &
495 & constants % SK_ri(l0, isph) * params % kappa
496 constants % C_ik(l0, isph) = one / (termi-termk)
497 end do
498 end do
499 if (params % fmm .eq. 1) then
500 tmp_pmax = max(params % pm, params % pl)
501 allocate(constants % SI_rnode(tmp_pmax+1, constants % nclusters), &
502 & constants % SK_rnode(tmp_pmax+1, constants % nclusters), &
503 & stat=info)
504 if (info.ne.0) then
505 constants % error_flag = 1
506 constants % error_message = "constants_init: `si_rnode, " // &
507 & "sk_rnode allocation failed"
508 return
509 end if
510 do i = 1, constants % nclusters
511 z = constants % rnode(i) * params % kappa
512 s1 = sqrt(two/(pi*real(z)))
513 s2 = sqrt(pi/(two*real(z)))
514 call cbesk(z, pt5, 1, 1, bessel_work(1), nz, ierr)
515 constants % SK_rnode(1, i) = s1 * real(bessel_work(1))
516 call cbesi(z, pt5, 1, 1, bessel_work(1), nz, ierr)
517 constants % SI_rnode(1, i) = s2 * real(bessel_work(1))
518 if (tmp_pmax .gt. 0) then
519 call cbesk(z, 1.5d0, 1, tmp_pmax, bessel_work(2:), nz, ierr)
520 constants % SK_rnode(2:, i) = s1 * real(bessel_work(2:tmp_pmax+1))
521 call cbesi(z, 1.5d0, 1, tmp_pmax, bessel_work(2:), nz, ierr)
522 constants % SI_rnode(2:, i) = s2 * real(bessel_work(2:tmp_pmax+1))
523 end if
524 end do
525 end if
526 deallocate(vylm, sk_rijn, dk_rijn, bessel_work, stat=info)
527 if (info .ne. 0) then
528 constants % error_flag = 1
529 constants % error_message = "constants_init: `vylm`, `SK_rijn` and " // &
530 & "`DK_rijn` deallocations failed"
531 return
532 end if
533 ! if doing incore build nonzero blocks of B
534 if (params % matvecmem .eq. 1) then
535 call build_b(constants, params)
536 end if
537 end if
538 deallocate(vplm, vcos, vsin, stat=info)
539 if (info .ne. 0) then
540 constants % error_flag = 1
541 constants % error_message = "constants_init: `vplm`, `vcos` and " // &
542 & "`vsin` deallocations failed"
543 return
544 end if
545 ! if doing incore build nonzero blocks of L
546 if (params % matvecmem .eq. 1) then
547 call build_itrnl(constants, params)
548 call build_l(constants, params)
549 end if
550end subroutine constants_init

578subroutine build_l(constants, params)
579 implicit none
580 type(ddx_params_type), intent(in) :: params
581 type(ddx_constants_type), intent(inout) :: constants
582 integer :: isph, ij, jsph, igrid, l, m, ind, info
583 real(dp), dimension(3) :: vij, sij
584 real(dp) :: vvij, tij, xij, oij, rho, ctheta, stheta, cphi, sphi, &
585 & fac, tt, thigh
586 real(dp), dimension(constants % nbasis) :: vylm, vplm
587 real(dp), dimension(params % lmax + 1) :: vcos, vsin
588 real(dp), dimension(constants % nbasis, params % ngrid) :: scratch
589 real(dp) :: t
590
591 allocate(constants % l(constants % nbasis, constants % nbasis, &
592 & constants % inl(params % nsph + 1)), stat=info)
593 if (info .ne. 0) then
594 constants % error_flag = 1
595 constants % error_message = 'Allocation failed in build_l'
596 return
597 end if
598
599 thigh = one + pt5*(params % se + one)*params % eta
600
601 t = omp_get_wtime()
602 !$omp parallel do default(none) shared(params,constants,thigh) &
603 !$omp private(isph,ij,jsph,scratch,igrid,vij,vvij,tij,sij,xij,oij, &
604 !$omp rho,ctheta,stheta,cphi,sphi,vylm,vplm,vcos,vsin,l,fac,ind,m,tt)
605 do isph = 1, params % nsph
606 do ij = constants % inl(isph), constants % inl(isph + 1) - 1
607 jsph = constants % nl(ij)
608 scratch = zero
609 do igrid = 1, params % ngrid
610 if (constants % ui(igrid, isph).eq.one) cycle
611 vij = params % csph(:, isph) &
612 & + params % rsph(isph)*constants % cgrid(:,igrid) &
613 & - params % csph(:, jsph)
614 vvij = sqrt(dot_product(vij, vij))
615 tij = vvij/params % rsph(jsph)
616 if (tij.lt.thigh .and. tij.gt.zero) then
617 sij = vij/vvij
618 xij = fsw(tij, params % se, params % eta)
619 if (constants % fi(igrid, isph).gt.one) then
620 oij = xij/constants % fi(igrid, isph)
621 else
622 oij = xij
623 end if
624 call ylmbas(sij, rho, ctheta, stheta, cphi, sphi, &
625 & params % lmax, constants % vscales, vylm, vplm, &
626 & vcos, vsin)
627 tt = oij
628 do l = 0, params % lmax
629 ind = l*l + l + 1
630 fac = - tt/(constants % vscales(ind)**2)
631 do m = -l, l
632 scratch(ind + m, igrid) = fac*vylm(ind + m)
633 end do
634 tt = tt*tij
635 end do
636 end if
637 end do
638 call dgemm('n', 't', constants % nbasis, constants % nbasis, params % ngrid, &
639 & one, constants % vwgrid, constants % vgrid_nbasis, scratch, &
640 & constants % nbasis, zero, constants % l(:,:,ij), constants % nbasis)
641 end do
642 end do
643end subroutine build_l

646subroutine build_b(constants, params)
647 implicit none
648 type(ddx_params_type), intent(in) :: params
649 type(ddx_constants_type), intent(inout) :: constants
650 integer :: isph, ij, jsph, igrid, l, m, ind
651 real(dp), dimension(3) :: vij, sij
652 real(dp) :: vvij, tij, xij, oij, rho, ctheta, stheta, cphi, sphi, &
653 & fac, vvtij, thigh
654 real(dp), dimension(constants % nbasis) :: vylm, vplm
655 real(dp), dimension(params % lmax + 1) :: vcos, vsin
656 complex(dp), dimension(max(2, params % lmax + 1)) :: bessel_work
657 real(dp), dimension(0:params % lmax) :: SI_rijn, DI_rijn
658 real(dp), dimension(constants % nbasis, params % ngrid) :: scratch
659 real(dp) :: t
660 integer :: info
661
662 allocate(constants % b(constants % nbasis, constants % nbasis, &
663 & constants % inl(params % nsph + 1)), stat=info)
664
665 thigh = one + pt5*(params % se + one)*params % eta
666
667 t = omp_get_wtime()
668 !$omp parallel do default(none) shared(params,constants,thigh) &
669 !$omp private(isph,ij,jsph,scratch,igrid,vij,vvij,tij,sij,xij,oij, &
670 !$omp rho,ctheta,stheta,cphi,sphi,vylm,vplm,vcos,vsin,si_rijn,di_rijn, &
671 !$omp vvtij,l,fac,ind,m,bessel_work)
672 do isph = 1, params % nsph
673 do ij = constants % inl(isph), constants % inl(isph + 1) - 1
674 jsph = constants % nl(ij)
675 scratch = zero
676 do igrid = 1, params % ngrid
677 if (constants % ui(igrid, isph).eq.one) cycle
678 vij = params % csph(:, isph) &
679 & + params % rsph(isph)*constants % cgrid(:,igrid) &
680 & - params % csph(:, jsph)
681 vvij = sqrt(dot_product(vij, vij))
682 tij = vvij/params % rsph(jsph)
683 if (tij.lt.thigh .and. tij.gt.zero) then
684 sij = vij/vvij
685 xij = fsw(tij, params % se, params % eta)
686 if (constants % fi(igrid, isph).gt.one) then
687 oij = xij/constants % fi(igrid, isph)
688 else
689 oij = xij
690 end if
691 call ylmbas(sij, rho, ctheta, stheta, cphi, sphi, &
692 & params % lmax, constants % vscales, vylm, vplm, &
693 & vcos, vsin)
694 si_rijn = 0
695 di_rijn = 0
696 vvtij = vvij*params % kappa
697 call modified_spherical_bessel_first_kind(params % lmax, &
698 & vvtij, si_rijn, di_rijn, bessel_work)
699 do l = 0, params % lmax
700 fac = - oij*si_rijn(l)/constants % SI_ri(l, jsph)
701 ind = l*l + l + 1
702 do m = -l, l
703 scratch(ind + m, igrid) = fac*vylm(ind + m)
704 end do
705 end do
706 end if

1275real(dp) function fsw(t, se, eta)
1276 ! Inputs
1277 real(dp), intent(in) :: t, se, eta
1278 ! Local variables
1279 real(dp) :: a, b, c, x
1280 ! Apply shift:
1281 ! se = 0 => t - eta/2 [ CENTERED ]
1282 ! se = 1 => t - eta [ EXTERIOR ]
1283 ! se = -1 => t [ INTERIOR ]
1284 x = t - (se*pt5 + pt5)*eta
1285 ! a must be in range (0,eta) for the switch region
1286 a = one - x
1287 ! Define switch function chi
1288 ! a <= 0 corresponds to the outside region
1289 if (a .le. zero) then
1290 fsw = zero
1291 ! a >= eta corresponds to the inside region
1292 else if (a .ge. eta) then
1293 fsw = one
1294 ! Switch region wher x is in range (1-eta,1)
1295 else
1296 ! Normalize a to region (0,1)
1297 a = a / eta
1298 b = 6d0*a - 15
1299 c = b*a + 10
1300 fsw = a * a * a * c
1301 end if
1302end function fsw

1330real(dp) function dfsw(t, se, eta)
1331 ! Inputs
1332 real(dp), intent(in) :: t, se, eta
1333 ! Local variables
1334 real(dp) :: flow, x
1335 real(dp), parameter :: f30=30.0d0
1336 ! Apply shift:
1337 ! s = 0 => t - eta/2 [ CENTERED ]
1338 ! s = 1 => t - eta [ EXTERIOR ]
1339 ! s = -1 => t [ INTERIOR ]
1340 x = t - (se + 1.d0)*eta / 2.d0
1341 ! Lower bound of switch region
1342 flow = one - eta
1343 ! Define derivative of switch function chi
1344 if (x .ge. one) then
1345 dfsw = zero
1346 else if (x .le. flow) then
1347 dfsw = zero
1348 else
1349 dfsw = -f30 * (( (one-x)*(x-one+eta) )**2) / (eta**5)
1350 endif
1351end function dfsw

1371 real(dp), allocatable :: work(:)
1372 external :: dgetrf, dgetri
1373 ! Allocation of temporaries
1374 allocate(ipiv(nbasis), work(nbasis**2), stat=info)
1375 if (info .ne. 0) then
1376 error_message = "mkprec: `ipiv` and `work` allocation failed"
1377 info = 1
1378 return
1379 endif
1380 ! Init
1381 rx_prc = zero
1382 ! Off-diagonal part
1383 do isph = 1, nsph
1384 do igrid = 1, ngrid
1385 f = twopi * ui(igrid, isph) * wgrid(igrid)
1386 do l1 = 0, lmax
1387 ind1 = l1*l1 + l1 + 1
1388 do m1 = -l1, l1
1389 f1 = f * vgrid(ind1+m1, igrid) / (two*dble(l1) + one)
1390 do lm = 1, nbasis
1391 rx_prc(lm, ind1+m1, isph) = &
1392 & rx_prc(lm, ind1+m1, isph) + f1*vgrid(lm, igrid)
1393 end do
1394 end do
1395 end do
1396 end do
1397 end do
1398 ! add diagonal
1399 f = twopi * (eps+one) / (eps-one)
1400 do isph = 1, nsph
1401 do lm = 1, nbasis
1402 rx_prc(lm, lm, isph) = rx_prc(lm, lm, isph) + f
1403 end do
1404 end do
1405 ! invert the blocks
1406 do isph = 1, nsph
1407 call dgetrf(nbasis, nbasis, rx_prc(:, :, isph), nbasis, ipiv, info)
1408 if (info .ne. 0) then
1409 error_message = "mkprec: dgetrf failed"
1410 info = 1
1411 return
1412 end if
1413 call dgetri(nbasis, rx_prc(:, :, isph), nbasis, ipiv, work, &
1414 & nbasis**2, info)
1415 if (info .ne. 0) then
1416 error_message = "mkprec: dgetri failed"
1417 info = 1
1418 return
1419 end if
1420 end do
1421 ! Cleanup temporaries
1422 deallocate(work, ipiv, stat=info)
1423 if (info .ne. 0) then
1424 error_message = "mkprec: `ipiv` and `work` deallocation failed"
1425 info = 1
1426 return
1427 end if
1428 ! Cleanup error message if there were no error
1429 error_message = ""
1430end subroutine mkprec

1453subroutine tree_rib_build(nsph, csph, rsph, order, cluster, children, parent, &
1454 & cnode, rnode, snode, error_message, error_flag)
1455 ! Inputs
1456 integer, intent(in) :: nsph
1457 real(dp), intent(in) :: csph(3, nsph), rsph(nsph)
1458 ! Outputs
1459 integer, intent(out) :: order(nsph), cluster(2, 2*nsph-1), &
1460 & children(2, 2*nsph-1), parent(2*nsph-1), snode(nsph)
1461 real(dp), intent(out) :: cnode(3, 2*nsph-1), rnode(2*nsph-1)
1462 integer, intent(out) :: error_flag
1463 character(len=255), intent(out) :: error_message
1464 ! Local variables
1465 integer :: nclusters, i, j, n, s, e, div
1466 real(dp) :: r, r1, r2, c(3), c1(3), c2(3), d, maxc
1467 !! At first construct the tree
1468 nclusters = 2*nsph - 1
1469 ! Init the root node
1470 cluster(1, 1) = 1
1471 cluster(2, 1) = nsph
1472 parent(1) = 0
1473 ! Index of the first unassigned node
1474 j = 2
1475 ! Init spheres ordering
1476 do i = 1, nsph
1477 order(i) = i
1478 end do
1479 ! Divide nodes until a single spheres
1480 do i = 1, nclusters
1481 ! The first sphere in i-th node
1482 s = cluster(1, i)
1483 ! The last sphere in i-th node
1484 e = cluster(2, i)
1485 ! Number of spheres in i-th node
1486 n = e - s + 1
1487 ! Divide only if there are 2 or more spheres
1488 if (n .gt. 1) then
1489 ! Use inertial bisection to reorder spheres and cut into 2 halfs
1490 call tree_rib_node_bisect(nsph, csph, n, order(s:e), div, &
1491 & error_message, error_flag)
1492 if (error_flag .ne. 0) return
1493 ! Assign the first half to the j-th node
1494 cluster(1, j) = s
1495 cluster(2, j) = s + div - 1
1496 ! Assign the second half to the (j+1)-th node
1497 cluster(1, j+1) = s + div
1498 cluster(2, j+1) = e
1499 ! Update list of children of i-th node
1500 children(1, i) = j
1501 children(2, i) = j + 1
1502 ! Set parents of new j-th and (j+1)-th node
1503 parent(j) = i
1504 parent(j+1) = i
1505 ! Shift index of the first unassigned node
1506 j = j + 2
1507 ! Set information for a leaf node
1508 else
1509 ! No children nodes
1510 children(:, i) = 0
1511 ! i-th node contains sphere ind(s)
1512 snode(order(s)) = i
1513 end if
1514 end do
1515 !! Compute bounding spheres of each node of the tree
1516 ! Bottom-to-top pass over all nodes of the tree
1517 do i = nclusters, 1, -1
1518 ! In case of a leaf node just use corresponding input sphere as a
1519 ! bounding sphere of the node
1520 if (children(1, i) .eq. 0) then
1521 ! Get correct index of the corresponding input sphere
1522 j = order(cluster(1, i))
1523 ! Get corresponding center and radius
1524 cnode(:, i) = csph(:, j)
1525 rnode(i) = rsph(j)
1526 ! In case of a non-leaf node get minimal sphere that contains bounding
1527 ! spheres of children nodes
1528 else
1529 ! The first child
1530 j = children(1, i)
1531 c1 = cnode(:, j)
1532 r1 = rnode(j)
1533 ! The second child
1534 j = children(2, i)
1535 c2 = cnode(:, j)
1536 r2 = rnode(j)
1537 ! Distance between centers of bounding spheres of children nodes
1538 c = c1 - c2
1539 ! Compute distance by scale and sum of scaled squares
1540 maxc = max(abs(c(1)), abs(c(2)), abs(c(3)))
1541 if (maxc .eq. zero) then
1542 d = zero
1543 else
1544 d = (c(1)/maxc)**2 + (c(2)/maxc)**2 + (c(3)/maxc)**2
1545 d = maxc * sqrt(d)
1546 end if
1547 ! If sphere #2 is completely inside sphere #1 use the first sphere
1548 if (r1 .ge. (r2+d)) then
1549 c = c1
1550 r = r1
1551 ! If sphere #1 is completely inside sphere #2 use the second sphere
1552 else if(r2 .ge. (r1+d)) then
1553 c = c2
1554 r = r2
1555 ! Otherwise use special formula to find a minimal sphere
1556 else
1557 r = (r1+r2+d) / 2
1558 c = c2 + c/d*(r-r2)
1559 end if
1560 ! Assign computed bounding sphere
1561 cnode(:, i) = c
1562 rnode(i) = r
1563 end if
1564 end do
1565end subroutine tree_rib_build

1576subroutine tree_rib_node_bisect(nsph, csph, n, order, div, error_message, &
1577 & error_flag)
1578 ! Inputs
1579 integer, intent(in) :: nsph, n
1580 real(dp), intent(in) :: csph(3, nsph)
1581 ! Outputs
1582 integer, intent(inout) :: order(n)
1583 integer, intent(out) :: div
1584 integer, intent(out) :: error_flag
1585 character(len=255), intent(out) :: error_message
1586 ! Local variables
1587 real(dp) :: c(3), s(3)
1588 real(dp), allocatable :: tmp_csph(:, :), work(:)
1589 external :: dgesvd
1590 integer :: i, l, r, lwork, info, istat
1591 integer, allocatable :: tmp_order(:)
1592
1593 allocate(tmp_csph(3, n), tmp_order(n), stat=istat)
1594 if (istat.ne.0) then
1595 error_message = 'Allocation failed in tree_node_bisect'
1596 error_flag = 1
1597 return
1598 end if
1599
1600 ! Get average coordinate
1601 c = zero
1602 do i = 1, n
1603 c = c + csph(:, order(i))
1604 end do
1605 c = c / n
1606 ! Get coordinates minus average in a contiguous array
1607 do i = 1, n
1608 tmp_csph(:, i) = csph(:, order(i)) - c
1609 end do
1610 !! Find right singular vectors
1611 ! Get proper size of temporary workspace
1612 lwork = -1
1613 call dgesvd('N', 'O', 3, n, tmp_csph, 3, s, tmp_csph, 3, tmp_csph, 3, &
1614 & s, lwork, info)
1615 lwork = int(s(1))
1616 allocate(work(lwork), stat=istat)
1617 if (istat.ne.0) then
1618 error_message = 'Allocation failed in tree_node_bisect'
1619 error_flag = 1
1620 return
1621 end if
1622 ! Get right singular vectors
1623 call dgesvd('N', 'O', 3, n, tmp_csph, 3, s, tmp_csph, 3, tmp_csph, 3, &
1624 & work, lwork, info)
1625 if (info.ne.0) then
1626 error_message = 'DGESVD failed in tree_node_bisect'
1627 error_flag = 1
1628 return
1629 end if
1630 deallocate(work, stat=istat)
1631 if (istat.ne.0) then
1632 error_message = 'Deallocation failed in tree_node_bisect'
1633 error_flag = 1
1634 return
1635 end if
1636 !! Sort spheres by sign of the above scalar product, which is equal to
1637 !! the leading right singular vector scaled by the leading singular value.
1638 !! However, we only care about sign, so we take into account only the
1639 !! leading right singular vector.
1640 ! First empty index from the left
1641 l = 1
1642 ! First empty index from the right
1643 r = n
1644 ! Cycle over values of the singular vector
1645 do i = 1, n
1646 ! Positive scalar products are moved to the beginning of `order`
1647 if (tmp_csph(1, i) .ge. zero) then
1648 tmp_order(l) = order(i)
1649 l = l + 1
1650 ! Negative scalar products are moved to the end of `order`
1651 else
1652 tmp_order(r) = order(i)
1653 r = r - 1
1654 end if
1655 end do
1656 ! Set divider and update order
1657 div = r
1658 order = tmp_order
1659 deallocate(tmp_csph, tmp_order, stat=istat)
1660 if (istat.ne.0) then
1661 error_message = 'Deallocation failed in tree_node_bisect'
1662 error_flag = 1
1663 return
1664 end if
1665end subroutine tree_rib_node_bisect

1691subroutine tree_get_farnear_work(n, children, cnode, rnode, lwork, iwork, &
1692 & jwork, work, nnfar, nfar, nnnear, nnear)
1693 ! Inputs
1694 integer, intent(in) :: n, children(2, n), lwork
1695 real(dp), intent(in) :: cnode(3, n), rnode(n)
1696 ! Outputs
1697 integer, intent(inout) :: iwork, jwork, work(3, lwork)
1698 integer, intent(out) :: nnfar, nfar(n), nnnear, nnear(n)
1699 ! Local variables
1700 integer :: j(2), npairs, k1, k2
1701 real(dp) :: c(3), r, d, dmax, dssq
1702 ! iwork is current temporary item in work array to process
1703 if (iwork .eq. 0) then
1704 work(1, 1) = 1
1705 work(2, 1) = 1
1706 iwork = 1
1707 jwork = 1
1708 end if
1709 ! jwork is total amount of temporary items in work array
1710 do while (iwork .le. jwork)
1711 j = work(1:2, iwork)
1712 c = cnode(:, j(1)) - cnode(:, j(2))
1713 r = rnode(j(1)) + rnode(j(2)) + max(rnode(j(1)), rnode(j(2)))
1714 !r = rnode(j(1)) + rnode(j(2))
1715 dmax = max(abs(c(1)), abs(c(2)), abs(c(3)))
1716 if (dmax .gt. 0) then
1717 dssq = (c(1)/dmax)**2 + (c(2)/dmax)**2 + (c(3)/dmax)**2
1718 else
1719 dssq = zero
1720 end if
1721 d = dmax * sqrt(dssq)
1722 !d = sqrt(c(1)**2 + c(2)**2 + c(3)**2)
1723 ! If node has no children then assume itself for purpose of finding
1724 ! far-field and near-filed interactions with children nodes of another
1725 ! node
1726 npairs = max(1, children(2, j(1))-children(1, j(1))+1) * &
1727 & max(1, children(2, j(2))-children(1, j(2))+1)
1728 if (d .ge. r) then
1729 ! Mark as far admissible pair
1730 work(3, iwork) = 1
1731 else if (npairs .eq. 1) then
1732 ! Mark as near admissible pair if both nodes are leaves
1733 work(3, iwork) = 2
1734 else if (jwork+npairs .gt. lwork) then
1735 ! Exit procedure, since work array was too small
1736 return
1737 else
1738 ! Mark as non-admissible pair and check all pairs of children nodes
1739 ! or pairs of one node (if it is a leaf node) with children of
1740 ! another node
1741 work(3, iwork) = 0
1742 if (children(1, j(1)) .eq. 0) then
1743 k1 = j(1)
1744 do k2 = children(1, j(2)), children(2, j(2))
1745 jwork = jwork + 1
1746 work(1, jwork) = k1
1747 work(2, jwork) = k2
1748 end do
1749 else if(children(1, j(2)) .eq. 0) then
1750 k2 = j(2)
1751 do k1 = children(1, j(1)), children(2, j(1))
1752 jwork = jwork + 1
1753 work(1, jwork) = k1
1754 work(2, jwork) = k2
1755 end do
1756 else
1757 do k1 = children(1, j(1)), children(2, j(1))
1758 do k2 = children(1, j(2)), children(2, j(2))
1759 jwork = jwork + 1
1760 work(1, jwork) = k1
1761 work(2, jwork) = k2
1762 end do
1763 end do
1764 end if
1765 end if
1766 iwork = iwork + 1
1767 end do
1768 nfar = 0
1769 nnear = 0
1770 do iwork = 1, jwork
1771 if (work(3, iwork) .eq. 1) then
1772 nfar(work(1, iwork)) = nfar(work(1, iwork)) + 1
1773 else if (work(3, iwork) .eq. 2) then
1774 nnear(work(1, iwork)) = nnear(work(1, iwork)) + 1
1775 end if
1776 end do
1777 iwork = jwork + 1
1778 nnfar = sum(nfar)
1779 nnnear = sum(nnear)
1780end subroutine tree_get_farnear_work

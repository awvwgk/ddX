159subroutine ddpcm(params, constants, workspace, state, phi_cav, gradphi_cav, &
160 & psi, tol, esolv, force)
161 implicit none
162 type(ddx_params_type), intent(in) :: params

215subroutine ddpcm_solve_worker(params, constants, workspace, phi_cav, &
216 & xs, xs_niter, xs_rel_diff, xs_time, tol, phi_grid, &
217 & phi, phiinf, phieps, phieps_niter, phieps_rel_diff, &
218 & phieps_time)
219 !! Inputs
220 type(ddx_params_type), intent(in) :: params
221 type(ddx_constants_type), intent(in) :: constants
222 real(dp), intent(in) :: phi_cav(constants % ncav), tol
223 !! Input+output
224 real(dp), intent(inout) :: xs(constants % nbasis, params % nsph), &
225 & phieps(constants % nbasis, params % nsph)
226 integer, intent(inout) :: xs_niter, phieps_niter
227 !! Temporary buffers
228 type(ddx_workspace_type), intent(inout) :: workspace
229 !! Outputs
230 real(dp), intent(out) :: phi_grid(params % ngrid, params % nsph), &
231 & phi(constants % nbasis, params % nsph), &
232 & phiinf(constants % nbasis, params % nsph), &
233 & xs_rel_diff(xs_niter), phieps_rel_diff(phieps_niter), xs_time, &
234 & phieps_time
235 !! Local variables
236 real(dp) :: start_time, finish_time, r_norm
237 character(len=255) :: string
238 real(dp), external :: ddot
239 !! The code
240 ! Unwrap sparsely stored potential at cavity points phi_cav into phi_grid
241 ! and multiply it by characteristic function at cavity points ui
242 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
243 & constants % icav_ia, constants % icav_ja, phi_cav, phi_grid)
244 workspace % tmp_cav = phi_cav * constants % ui_cav
245 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
246 & constants % icav_ia, constants % icav_ja, workspace % tmp_cav, &
247 & workspace % tmp_grid)
248 ! Integrate against spherical harmonics and Lebedev weights to get Phi
249 call ddintegrate_sph_work(constants % nbasis, params % ngrid, &
250 & params % nsph, constants % vwgrid, constants % vgrid_nbasis, &
251 & one, workspace % tmp_grid, zero, phi)
252 ! Compute Phi_infty
253 ! force dx called from rinfx to add the diagonal
254 call rinfx(params, constants, workspace, phi, phiinf)
255 ! Select initial guess for the ddPCM system
256 call cpu_time(start_time)
257 call jacobi_diis(params, constants, workspace, tol, phiinf, phieps, &
258 & phieps_niter, phieps_rel_diff, repsx, prec_repsx, hnorm)
259 call cpu_time(finish_time)
260 phieps_time = finish_time - start_time
261 ! Check if solver did not converge
262 if (workspace % error_flag .ne. 0) then
263 workspace % error_message = "ddpcm_energy: solver for ddPCM system " // &
264 & " did not converge"
265 return
266 end if
267 ! Set right hand side to -Phi_epsilon
268 workspace % tmp_rhs = -phieps
269 ! Solve ddCOSMO system L X = -Phi_epsilon with a proper initial guess
270 call cpu_time(start_time)
271 call jacobi_diis(params, constants, workspace, tol, workspace % tmp_rhs, &
272 & xs, xs_niter, xs_rel_diff, lx, ldm1x, hnorm)
273 call cpu_time(finish_time)
274 xs_time = finish_time - start_time
275 ! Check if solver did not converge
276 if (workspace % error_flag .ne. 0) then
277 workspace % error_message = "ddpcm_energy: solver for ddCOSMO system " // &
278 & " did not converge"
279 return
280 end if
281 ! Clear status
282end subroutine ddpcm_solve_worker

415subroutine ddpcm_forces_worker(params, constants, workspace, phi_grid, &
416 & gradphi_cav, psi, phi, phieps, s, sgrid, y, ygrid, g, q, qgrid, xs, &
417 & zeta, force)
418 !! Inputs
419 type(ddx_params_type), intent(in) :: params
420 type(ddx_constants_type), intent(in) :: constants
421 real(dp), intent(in) :: phi_grid(params % ngrid, params % nsph), &
422 & gradphi_cav(3, constants % ncav), &
423 & psi(constants % nbasis, params % nsph), &
424 & phi(constants % nbasis, params % nsph), &
425 & phieps(constants % nbasis, params % nsph), &
426 & s(constants % nbasis, params % nsph), &
427 & y(constants % nbasis, params % nsph), &
428 & xs(constants % nbasis, params % nsph)
429 !! Temporary buffers
430 type(ddx_workspace_type), intent(inout) :: workspace
431 !! Outputs
432 real(dp), intent(out) :: sgrid(params % ngrid, params % nsph), &
433 & ygrid(params % ngrid, params % nsph), &
434 & g(constants % nbasis, params % nsph), &
435 & q(constants % nbasis, params % nsph), &
436 & qgrid(params % ngrid, params % nsph), &
437 & zeta(constants % ncav), &
438 & force(3, params % nsph)
439 !! Local variables
440 integer :: isph, icav, igrid, inode, jnode, jsph, jnear
441 real(dp) :: tmp1, tmp2, d(3), dnorm
442 real(dp), external :: ddot, dnrm2
443 !! The code
444 ! Get grid values of S and Y
445 call dgemm('T', 'N', params % ngrid, params % nsph, &
446 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
447 & s, constants % nbasis, zero, sgrid, params % ngrid)
448 call dgemm('T', 'N', params % ngrid, params % nsph, &
449 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
450 & y, constants % nbasis, zero, ygrid, params % ngrid)
451 g = phieps - phi
452 q = s - fourpi/(params % eps-one)*y
453 qgrid = sgrid - fourpi/(params % eps-one)*ygrid
454 ! gradr initializes forces with zeros
455 call gradr(params, constants, workspace, g, ygrid, force)
456 do isph = 1, params % nsph
457 call contract_grad_l(params, constants, isph, xs, sgrid, &
458 & workspace % tmp_vylm(:, 1), workspace % tmp_vdylm(:, :, 1), &
459 & workspace % tmp_vplm(:, 1), workspace % tmp_vcos(:, 1), &
460 & workspace % tmp_vsin(:, 1), force(:, isph))
461 call contract_grad_u(params, constants, isph, qgrid, phi_grid, force(:, isph))
462 end do
463 force = -pt5 * force
464 icav = 0
465 do isph = 1, params % nsph
466 do igrid = 1, params % ngrid
467 if(constants % ui(igrid, isph) .ne. zero) then
468 icav = icav + 1
469 zeta(icav) = constants % wgrid(igrid) * &
470 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
471 & constants % vgrid(1, igrid), 1, q(1, isph), 1)
472 force(:, isph) = force(:, isph) &
473 & - pt5*zeta(icav)*gradphi_cav(:, icav)
474 end if
475 end do
476 end do
477 !! Last term where we compute gradients of potential at centers of atoms
478 !! spawned by intermediate zeta.
479 if(params % fmm .eq. 1) then
480 !! This step can be substituted by a proper dgemm if zeta
481 !! intermediate is converted from cavity points to all grid points
482 !! with zero values at internal grid points
483 ! P2M step
484 icav = 0
485 do isph = 1, params % nsph
486 inode = constants % snode(isph)
487 workspace % tmp_node_m(:, inode) = zero
488 do igrid = 1, params % ngrid
489 if(constants % ui(igrid, isph) .eq. zero) cycle
490 icav = icav + 1
491 call fmm_p2m(constants % cgrid(:, igrid), zeta(icav), &
492 & one, params % pm, constants % vscales, one, &
493 & workspace % tmp_node_m(:, inode))
494 end do
495 workspace % tmp_node_m(:, inode) = workspace % tmp_node_m(:, inode) / &
496 & params % rsph(isph)
497 end do
498 ! M2M, M2L and L2L translations
499 call tree_m2m_rotation(params, constants, workspace % tmp_node_m)
500 call tree_m2l_rotation(params, constants, workspace % tmp_node_m, &
501 & workspace % tmp_node_l)
502 call tree_l2l_rotation(params, constants, workspace % tmp_node_l)
503 ! Now compute near-field FMM gradients
504 ! Cycle over all spheres
505 icav = 0
506 workspace % tmp_efld = zero
507 do isph = 1, params % nsph
508 ! Cycle over all external grid points
509 do igrid = 1, params % ngrid
510 if(constants % ui(igrid, isph) .eq. zero) cycle
511 icav = icav + 1
512 ! Cycle over all near-field admissible pairs of spheres,
513 ! including pair (isph, isph) which is a self-interaction
514 inode = constants % snode(isph)
515 do jnear = constants % snear(inode), constants % snear(inode+1)-1
516 ! Near-field interactions are possible only between leaf
517 ! nodes, which must contain only a single input sphere
518 jnode = constants % near(jnear)
519 jsph = constants % order(constants % cluster(1, jnode))
520 d = params % csph(:, isph) + &
521 & constants % cgrid(:, igrid)*params % rsph(isph) - &
522 & params % csph(:, jsph)
523 dnorm = dnrm2(3, d, 1)
524 workspace % tmp_efld(:, jsph) = workspace % tmp_efld(:, jsph) + &
525 & zeta(icav)*d/(dnorm**3)
526 end do
527 end do
528 end do
529 ! Take into account far-field FMM gradients only if pl > 0
530 if (params % pl .gt. 0) then
531 tmp1 = one / sqrt(3d0) / constants % vscales(1)
532 do isph = 1, params % nsph
533 inode = constants % snode(isph)
534 tmp2 = tmp1 / params % rsph(isph)
535 workspace % tmp_efld(3, isph) = workspace % tmp_efld(3, isph) + &
536 & tmp2*workspace % tmp_node_l(3, inode)
537 workspace % tmp_efld(1, isph) = workspace % tmp_efld(1, isph) + &
538 & tmp2*workspace % tmp_node_l(4, inode)
539 workspace % tmp_efld(2, isph) = workspace % tmp_efld(2, isph) + &
540 & tmp2*workspace % tmp_node_l(2, inode)
541 end do
542 end if
543 do isph = 1, params % nsph
544 force(:, isph) = force(:, isph) &
545 & - pt5*workspace % tmp_efld(:, isph)*params % charge(isph)
546 end do
547 ! Naive quadratically scaling implementation
548 else
549 ! This routines actually computes -grad, not grad
550 call efld(constants % ncav, zeta, constants % ccav, params % nsph, &
551 & params % csph, workspace % tmp_efld)
552 do isph = 1, params % nsph
553 force(:, isph) = force(:, isph) &
554 & + pt5*workspace % tmp_efld(:, isph)*params % charge(isph)
555 end do
556 end if
557 ! Clear status
558end subroutine ddpcm_forces_worker

560subroutine ddpcm_geom_forces_worker(params, constants, workspace, phi_grid, &
561 & gradphi_cav, psi, phi, phieps, s, sgrid, y, ygrid, g, q, qgrid, xs, &
562 & zeta, force)
563 !! Inputs
564 type(ddx_params_type), intent(in) :: params
565 type(ddx_constants_type), intent(in) :: constants
566 real(dp), intent(in) :: phi_grid(params % ngrid, params % nsph), &
567 & gradphi_cav(3, constants % ncav), &
568 & psi(constants % nbasis, params % nsph), &
569 & phi(constants % nbasis, params % nsph), &
570 & phieps(constants % nbasis, params % nsph), &
571 & s(constants % nbasis, params % nsph), &
572 & y(constants % nbasis, params % nsph), &
573 & xs(constants % nbasis, params % nsph)
574 !! Temporary buffers
575 type(ddx_workspace_type), intent(inout) :: workspace
576 !! Outputs
577 real(dp), intent(out) :: sgrid(params % ngrid, params % nsph), &
578 & ygrid(params % ngrid, params % nsph), &
579 & g(constants % nbasis, params % nsph), &
580 & q(constants % nbasis, params % nsph), &
581 & qgrid(params % ngrid, params % nsph), &
582 & zeta(constants % ncav), &
583 & force(3, params % nsph)
584 !! Local variables
585 integer :: isph, icav, igrid, inode, jnode, jsph, jnear
586 real(dp) :: tmp1, tmp2, d(3), dnorm
587 real(dp), external :: ddot, dnrm2

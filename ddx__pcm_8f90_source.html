39subroutine ddpcm(params, constants, workspace, state, phi_cav, &
40 & psi, tol, esolv, force)
41 implicit none
42 type(ddx_params_type), intent(in) :: params
43 type(ddx_constants_type), intent(in) :: constants
44 type(ddx_workspace_type), intent(inout) :: workspace
45 type(ddx_state_type), intent(inout) :: state
46 real(dp), intent(in) :: phi_cav(constants % ncav), &
47 & psi(constants % nbasis, params % nsph), tol
48 real(dp), intent(out) :: esolv, force(3, params % nsph)
49 real(dp), external :: ddot
50
51 call ddpcm_setup(params, constants, workspace, state, phi_cav, psi)
52 call ddpcm_guess(params, constants, workspace, state)
53 call ddpcm_solve(params, constants, workspace, state, tol)
54
55 ! Compute the solvation energy
56 esolv = pt5*ddot(constants % n, state % xs, 1, psi, 1)
57
58 ! Get forces if needed
59 if (params % force .eq. 1) then
60 ! solve the adjoint
61 call ddpcm_guess_adjoint(params, constants, workspace, state)
62 call ddpcm_solve_adjoint(params, constants, workspace, state, tol)
63
64 ! evaluate the solvent unspecific contribution analytical derivatives
65 force = zero
66 call ddpcm_solvation_force_terms(params, constants, workspace, &
67 & state, force)
68 end if

246subroutine ddpcm_solvation_force_terms(params, constants, workspace, &
247 & state, force)
248 implicit none
249 type(ddx_params_type), intent(in) :: params
250 type(ddx_constants_type), intent(in) :: constants
251 type(ddx_workspace_type), intent(inout) :: workspace
252 type(ddx_state_type), intent(inout) :: state
253 real(dp), intent(out) :: force(3, params % nsph)
254 ! local
255 real(dp) :: start_time, finish_time
256 real(dp), external :: ddot
257 integer :: icav, isph, igrid
258
259 start_time = omp_get_wtime()
260 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
261 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
262 & state % phi_grid)
263 ! Get grid values of S and Y
264 call dgemm('T', 'N', params % ngrid, params % nsph, &
265 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
266 & state % s, constants % nbasis, zero, state % sgrid, params % ngrid)
267 call dgemm('T', 'N', params % ngrid, params % nsph, &
268 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
269 & state % y, constants % nbasis, zero, state % ygrid, params % ngrid)
270
271 state % g = - (state % phieps - state % phi)
272 state % q = state % s - fourpi/(params % eps - one)*state % y
273 state % qgrid = state % sgrid - fourpi/(params % eps - one)*state % ygrid
274
275 call gradr(params, constants, workspace, state % g, state % ygrid, force)
276
277 do isph = 1, params % nsph
278 call contract_grad_l(params, constants, isph, state % xs, &
279 & state % sgrid, workspace % tmp_vylm(:, 1), &
280 & workspace % tmp_vdylm(:, :, 1), workspace % tmp_vplm(:, 1), &
281 & workspace % tmp_vcos(:, 1), workspace % tmp_vsin(:, 1), &
282 & force(:, isph))
283 call contract_grad_u(params, constants, isph, state % qgrid, &
284 & state % phi_grid, force(:, isph))
285 end do
286 force = - pt5 * force
287
288 icav = 0
289 do isph = 1, params % nsph
290 do igrid = 1, params % ngrid
291 if(constants % ui(igrid, isph) .ne. zero) then
292 icav = icav + 1
293 state % zeta(icav) = constants % wgrid(igrid) * &
294 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
295 & constants % vgrid(1, igrid), 1, &
296 & state % q(1, isph), 1)
297 end if
298 end do
299 end do
300 finish_time = omp_get_wtime()
301 state % force_time = finish_time - start_time
302
303end subroutine ddpcm_solvation_force_terms

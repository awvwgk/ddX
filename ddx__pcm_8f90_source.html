39subroutine ddpcm(params, constants, workspace, state, phi_cav, &
40 & psi, tol, esolv, force)
41 implicit none
42 type(ddx_params_type), intent(in) :: params
43 type(ddx_constants_type), intent(in) :: constants
44 type(ddx_workspace_type), intent(inout) :: workspace
45 type(ddx_state_type), intent(inout) :: state
46 real(dp), intent(in) :: phi_cav(constants % ncav), &
47 & psi(constants % nbasis, params % nsph), tol
48 real(dp), intent(out) :: esolv
49 real(dp), intent(out), optional :: force(3, params % nsph)
50
51 call ddpcm_setup(params, constants, workspace, state, phi_cav, psi)
52 call ddpcm_guess(params, constants, workspace, state)
53 call ddpcm_solve(params, constants, workspace, state, tol)
54
55 call ddpcm_energy(constants, state, esolv)
56
57 ! Get forces if needed
58 if (params % force .eq. 1) then
59 ! solve the adjoint
60 call ddpcm_guess_adjoint(params, constants, workspace, state)
61 call ddpcm_solve_adjoint(params, constants, workspace, state, tol)
62
63 ! evaluate the solvent unspecific contribution analytical derivatives
64 force = zero
65 call ddpcm_solvation_force_terms(params, constants, workspace, &
66 & state, force)
67 end if

212subroutine ddpcm_solve_adjoint(params, constants, workspace, state, tol)
213 implicit none
214 type(ddx_params_type), intent(in) :: params
215 type(ddx_constants_type), intent(in) :: constants
216 type(ddx_workspace_type), intent(inout) :: workspace
217 type(ddx_state_type), intent(inout) :: state
218 real(dp), intent(in) :: tol
219 ! local variables
220 real(dp) :: start_time, finish_time
221
222 state % s_niter = params % maxiter
223 start_time = omp_get_wtime()
224 call jacobi_diis(params, constants, workspace, tol, state % psi, &
225 & state % s, state % s_niter, state % s_rel_diff, lstarx, ldm1x, &
226 & hnorm)
227 finish_time = omp_get_wtime()
228 state % s_time = finish_time - start_time
229
230 if (workspace % error_flag .ne. 0) then
231 workspace % error_message = "ddpcm_energy: solver for adjoint " // &
232 & "ddCOSMO system did not converge"
233 return
234 end if
235
236 state % y_niter = params % maxiter
237 start_time = omp_get_wtime()
238 call jacobi_diis(params, constants, workspace, tol, state % s, state % y, &
239 & state % y_niter, state % y_rel_diff, repsstarx, prec_repsstarx, &
240 & hnorm)
241 finish_time = omp_get_wtime()
242 state % y_time = finish_time - start_time
243
244 if (workspace % error_flag .ne. 0) then
245 workspace % error_message = "ddpcm_energy: solver for adjoint " // &
246 & "ddPCM system did not converge"
247 return
248 end if
249
250 ! compute the real adjoint solution and store it in Q
251 state % q = state % s - fourpi/(params % eps - one)*state % y
252
253 call ddpcm_derivative_setup(params, constants, workspace, state)
254

266subroutine ddpcm_solvation_force_terms(params, constants, workspace, &
267 & state, force)
268 implicit none
269 type(ddx_params_type), intent(in) :: params
270 type(ddx_constants_type), intent(in) :: constants
271 type(ddx_workspace_type), intent(inout) :: workspace
272 type(ddx_state_type), intent(inout) :: state
273 real(dp), intent(out) :: force(3, params % nsph)
274
275 real(dp) :: start_time, finish_time
276 integer :: isph
277
278 start_time = omp_get_wtime()
279
280 call gradr(params, constants, workspace, state % g, state % ygrid, force)
281
282 do isph = 1, params % nsph
283 call contract_grad_l(params, constants, isph, state % xs, &
284 & state % sgrid, workspace % tmp_vylm(:, 1), &
285 & workspace % tmp_vdylm(:, :, 1), workspace % tmp_vplm(:, 1), &
286 & workspace % tmp_vcos(:, 1), workspace % tmp_vsin(:, 1), &
287 & force(:, isph))

289 & state % phi_grid, force(:, isph))
290 end do
291 force = - pt5 * force
292
293 finish_time = omp_get_wtime()
294 state % force_time = finish_time - start_time

306subroutine ddpcm_derivative_setup(params, constants, workspace, state)
307 type(ddx_params_type), intent(in) :: params
308 type(ddx_constants_type), intent(in) :: constants
309 type(ddx_workspace_type), intent(inout) :: workspace
310 type(ddx_state_type), intent(inout) :: state
311
312 real(dp), external :: ddot
313 integer :: icav, isph, igrid
314
315 ! Get grid values of S and Y
316 call dgemm('T', 'N', params % ngrid, params % nsph, &
317 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
318 & state % s, constants % nbasis, zero, state % sgrid, params % ngrid)
319 call dgemm('T', 'N', params % ngrid, params % nsph, &
320 & constants % nbasis, one, constants % vgrid, constants % vgrid_nbasis, &
321 & state % y, constants % nbasis, zero, state % ygrid, params % ngrid)
322
323 ! Get the values of phi on the grid
324 call ddcav_to_grid_work(params % ngrid, params % nsph, constants % ncav, &
325 & constants % icav_ia, constants % icav_ja, state % phi_cav, &
326 & state % phi_grid)
327
328 state % g = - (state % phieps - state % phi)
329 state % qgrid = state % sgrid - fourpi/(params % eps - one)*state % ygrid
330
331 ! assemble the intermediate zeta: S weighted by U evaluated on the
332 ! exposed grid points.
333 icav = 0
334 do isph = 1, params % nsph
335 do igrid = 1, params % ngrid
336 if (constants % ui(igrid, isph) .ne. zero) then
337 icav = icav + 1
338 state % zeta(icav) = constants % wgrid(igrid) * &
339 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
340 & constants % vgrid(1, igrid), 1, state % q(1, isph), 1)
341 end if
342 end do
343 end do
344
345end subroutine ddpcm_derivative_setup

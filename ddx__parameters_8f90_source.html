147subroutine params_init(model, force, eps, kappa, eta, se, lmax, ngrid, &

151 !! Inputs
152 ! Model to use 1 for COSMO, 2 for PCM, 3 for LPB.
153 integer, intent(in) :: model
154 ! Whether computing analytical forces will be required (1) or not (0).
155 integer, intent(in) :: force
156 ! Relative dielectric permittivity.
157 real(dp), intent(in) :: eps
158 ! Debye-H\"{u}ckel parameter. Referenced only in LPB model (model=3)
159 real(dp), intent(in) :: kappa
160 ! Regularization parameter.
161 real(dp), intent(in) :: eta
162 ! Shift of the regularization. -1 for interior, 0 for centered and
163 ! 1 for outer regularization.
164 real(dp), intent(in) :: se
165 ! Maximal degree of modeling spherical harmonics.
166 integer, intent(in) :: lmax
167 ! Number of Lebedev grid points on each sphere.
168 integer, intent(in) :: ngrid
169 ! handling of sparse matrix. 1 for precomputing them and keeping them in
170 ! memory, 0 for assembling the mvps on-the-fly.
171 integer, intent(in) :: matvecmem
172 ! Iterative solver to be used. 1 for Jacobi/DIIS, 2 for GMRES.
173 ! Other solvers might be added later.
174 integer, intent(in) :: itersolver
175 ! Maximum number of iterations for the iterative solver.
176 integer, intent(in) :: maxiter
177 ! Number of extrapolation points for Jacobi/DIIS solver. Referenced only
178 ! if Jacobi solver is used.
179 integer, intent(in) :: jacobi_ndiis
180 ! Number of last vectors that GMRESR works with. Referenced only if GMRESR
181 ! solver is used.
182 integer, intent(in) :: gmresr_j
183 ! Dimension of the envoked GMRESR. In case of 0 GMRESR becomes the GCR
184 ! solver, one of the simplest versions of GMRESR. Referenced only if
185 ! GMRESR solver is used.
186 integer, intent(in) :: gmresr_dim
187 ! Enable (1) or disable (0) use of FMM techniques.
188 integer, intent(in) :: fmm
189 ! Maximal degree of spherical harmonics for a multipole expansion.
190 !
191 ! If this value is -1 then no far-field FMM interactions are performed.
192 integer, intent(in) :: pm
193 ! Maximal degree of spherical harmonics for a local expansion.
194 !
195 ! If this value is -1 then no far-field FMM interactions are performed.
196 integer, intent(in) :: pl
197 ! Number of OpenMP threads to be used.
198 integer, intent(in) :: nproc
199 ! Number of atoms in the molecule.
200 integer, intent(in) :: nsph
201 ! Charges of atoms of a dimension (nsph).
202 real(dp), intent(in) :: charge(nsph)
203 ! Centers of atoms of a dimension (3, nsph).
204 real(dp), intent(in) :: csph(3, nsph)
205 ! Array of radii of atoms of a dimension (nsph).
206 real(dp), intent(in) :: rsph(nsph)
207 ! Error printing function
208 procedure(print_func_interface) :: print_func

211 integer, intent(out) :: info

213 integer :: igrid, i

215 !! The code
216 ! Model, 1=COSMO, 2=PCM, 3=LPB
217 if ((model .lt. 1) .or. (model .gt. 3)) then
218 params % error_flag = 1
219 params % error_message = "params_init: invalid value of `model`"
220 call print_func(params % error_message)
221 info = -1
222 return
223 end if
224 params % model = model
225 ! Check if forces are needed
226 if ((force .lt. 0) .or. (force .gt. 1)) then
227 params % error_flag = 1
228 params % error_message = "params_init: invalid value of `force`"
229 call print_func(params % error_message)
230 info = -1
231 return
232 end if
233 params % force = force
234 ! Relative dielectric permittivity
235 if (eps .le. one) then
236 params % error_flag = 1
237 params % error_message = "params_init: invalid value of `eps`"
238 call print_func(params % error_message)
239 info = -1
240 return
241 end if
242 params % eps = eps
243 ! Debye-H\"{u}ckel parameter (only used in ddLPB)
244 if ((model .eq. 3) .and. (kappa .le. zero)) then
245 params % error_flag = 1
246 params % error_message = "params_init: invalid value of `kappa`"
247 call print_func(params % error_message)
248 info = -1
249 return
250 end if
251 params % kappa = kappa
252 ! Regularization parameter
253 if ((eta .lt. zero) .or. (eta .gt. one)) then
254 params % error_flag = 1
255 params % error_message = "params_init: invalid value of `eta`"
256 call print_func(params % error_message)
257 info = -1
258 return
259 end if
260 params % eta = eta
261 ! Shift of a regularization
262 if ((se .lt. -one) .or. (se .gt. one)) then
263 params % error_flag = 1
264 params % error_message = "params_init: invalid value of `se`"
265 call print_func(params % error_message)
266 info = -1
267 return
268 end if
269 params % se = se
270 ! Degree of modeling spherical harmonics
271 if (lmax .lt. 0) then
272 params % error_flag = 1
273 params % error_message = "params_init: invalid value of `lmax`"
274 call print_func(params % error_message)
275 info = -1
276 return
277 end if
278 params % lmax = lmax
279 ! Check number of Lebedev grid points
280 igrid = 0
281 do i = 1, nllg
282 if (ng0(i) .eq. ngrid) then
283 igrid = i
284 exit
285 end if
286 end do
287 if (igrid .eq. 0) then
288 params % error_flag = 1
289 params % error_message = "params_init: Unsupported value of `ngrid`"
290 call print_func(params % error_message)
291 info = -1
292 return
293 end if
294 params % ngrid = ngrid
295 ! Iterative solver: 1=Jacobi 2=GMRES
296 if ((itersolver .lt. 1) .or. (itersolver .gt. 2)) then
297 params % error_flag = 1
298 params % error_message = "params_init: invalid value of `itersolver`"
299 call print_func(params % error_message)
300 info = -1
301 return
302 end if
303 params % itersolver = itersolver
304 ! Maximum number of iterations
305 if (maxiter .le. 0) then
306 params % error_flag = 1
307 params % error_message = "params_init: invalid value of `maxiter`"
308 call print_func(params % error_message)
309 info = -1
310 return
311 end if
312 params % maxiter = maxiter
313 ! Number of Jacobi DIIS extrapolation points (ndiis=25 works)
314 if (jacobi_ndiis .lt. 0) then
315 params % error_flag = 1
316 params % error_message = "params_init: invalid value of `jacobi_ndiis`"
317 call print_func(params % error_message)
318 info = -1
319 return
320 end if
321 params % jacobi_ndiis = jacobi_ndiis
322 ! Number of last vectors that GMRESR works with. Referenced only if GMRESR
323 ! solver is used.
324 if (gmresr_j .lt. 1) then
325 params % error_flag = 1
326 params % error_message = "params_init: invalid value of `gmresr_j`"
327 call print_func(params % error_message)
328 info = -1
329 return
330 end if
331 params % gmresr_j = gmresr_j
332 ! Dimension of the envoked GMRESR. In case of 0 GMRESR becomes the GCR
333 ! solver, one of the simplest versions of GMRESR. Referenced only if
334 ! GMRESR solver is used.
335 if (gmresr_dim .lt. 0) then
336 params % error_flag = 1
337 params % error_message = "params_init: invalid value of `gmresr_j`"
338 call print_func(params % error_message)
339 info = -1
340 return
341 end if
342 params % gmresr_dim = gmresr_dim
343 ! Check if FMM-acceleration is needed
344 if ((fmm .lt. 0) .or. (fmm .gt. 1)) then
345 params % error_flag = 1
346 params % error_message = "params_init: invalid value of `fmm`"
347 call print_func(params % error_message)
348 info = -1
349 return
350 end if
351 params % fmm = fmm
352 ! Set FMM parameters if FMM is needed
353 if (fmm .eq. 1) then
354 ! Maximal degree of multipole spherical harmonics. Value -1 means no
355 ! far-field interactions are to be computed, only near-field
356 ! interactions are taken into account.
357 if (pm .lt. -1) then
358 params % error_flag = 1
359 params % error_message = "params_init: invalid value of `pm`"
360 call print_func(params % error_message)
361 info = -1
362 return
363 end if
364 ! Maximal degree of local spherical harmonics. Value -1 means no
365 ! far-field interactions are to be computed, only near-field
366 ! interactions are taken into account.
367 if (pl .lt. -1) then
368 params % error_flag = 1
369 params % error_message = "params_init: invalid value of `pl`"
370 call print_func(params % error_message)
371 info = -1
372 return
373 end if
374 ! If far-field interactions are to be ignored
375 if ((pl .eq. -1) .or. (pm .eq. -1)) then
376 params % pm = -1
377 params % pl = -1
378 ! If far-field interactions are to be taken into account
379 else
380 params % pm = pm
381 params % pl = pl
382 end if
383 else
384 ! These values are ignored if fmm flag is 0
385 params % pm = -2
386 params % pl = -2
387 end if
388 ! Number of OpenMP threads to be used
389 ! available.
390 if (nproc .lt. 0) then
391 params % error_flag = 1
392 params % error_message = "params_init: invalid value of `nproc`"
393 call print_func(params % error_message)
394 info = -1
395 return
396 else if (nproc .eq. 0) then
397 params % nproc = 1
398 else
399 params % nproc = nproc
400 end if
401 call omp_set_num_threads(params % nproc)
402 ! Number of atoms
403 if (nsph .le. 0) then
404 params % error_flag = 1
405 params % error_message = "params_init: invalid value of `nsph`"
406 call print_func(params % error_message)
407 info = -1
408 return
409 end if
410 params % nsph = nsph
411 ! Allocation of charge, csph and rsph
412 allocate(params % charge(nsph), params % csph(3, nsph), &
413 & params % rsph(nsph), stat=info)
414 if (info .ne. 0) then
415 params % error_flag = 1
416 params % error_message = "params_init: `charge`, `csph` and `rsph` " &
417 & // "allocations failed"
418 call print_func(params % error_message)
419 info = 1
420 return
421 end if
422 params % charge = charge
423 params % csph = csph
424 params % rsph = rsph
425 if (matvecmem.eq.0 .or. matvecmem.eq.1) then
426 params % matvecmem = matvecmem
427 else
428 params % error_flag = 1
429 params % error_message = "params_init: invalid value of `matvecmem`"
430 call print_func(params % error_message)
431 info = -1
432 return
433 end if
434 ! Set print function for errors
435 params % print_func => print_func
436 ! Clear error state
437 info = 0
438 params % error_flag = 0
439 params % error_message = ""
440end subroutine params_init

547subroutine closest_supported_lebedev_grid(ngrid)
548 integer, intent(inout) :: ngrid
549 integer :: igrid, i, inear, jnear
550 ! Get nearest number of Lebedev grid points
551 igrid = 0
552 inear = 100000
553 do i = 1, nllg
554 jnear = abs(ng0(i) - ngrid)
555 if (jnear .lt. inear) then
556 inear = jnear
557 igrid = i
558 end if
559 end do
560 ngrid = ng0(igrid)
561end subroutine

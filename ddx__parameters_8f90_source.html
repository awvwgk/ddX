143subroutine params_init(model, force, eps, kappa, eta, se, lmax, ngrid, &
144 & matvecmem, maxiter, jacobi_ndiis, &
145 & fmm, pm, pl, nproc, nsph, charge, &
146 & csph, rsph, print_func, output_filename, params)
147 !! Inputs
148 ! Model to use 1 for COSMO, 2 for PCM, 3 for LPB.
149 integer, intent(in) :: model
150 ! Whether computing analytical forces will be required (1) or not (0).
151 integer, intent(in) :: force
152 ! Relative dielectric permittivity.
153 real(dp), intent(in) :: eps
154 ! Debye-H\"{u}ckel parameter. Referenced only in LPB model (model=3)
155 real(dp), intent(in) :: kappa
156 ! Regularization parameter.
157 real(dp), intent(in) :: eta
158 ! Shift of the regularization. -1 for interior, 0 for centered and
159 ! 1 for outer regularization.
160 real(dp), intent(in) :: se
161 ! Maximal degree of modeling spherical harmonics.
162 integer, intent(in) :: lmax
163 ! Number of Lebedev grid points on each sphere.
164 integer, intent(in) :: ngrid
165 ! handling of sparse matrix. 1 for precomputing them and keeping them in
166 ! memory, 0 for assembling the mvps on-the-fly.
167 integer, intent(in) :: matvecmem
168 ! Maximum number of iterations for the iterative solver.
169 integer, intent(in) :: maxiter
170 ! Number of extrapolation points for Jacobi/DIIS solver. Referenced only
171 ! if Jacobi solver is used.
172 integer, intent(in) :: jacobi_ndiis
173 ! Enable (1) or disable (0) use of FMM techniques.
174 integer, intent(in) :: fmm
175 ! Maximal degree of spherical harmonics for a multipole expansion.
176 !
177 ! If this value is -1 then no far-field FMM interactions are performed.
178 integer, intent(in) :: pm
179 ! Maximal degree of spherical harmonics for a local expansion.
180 !
181 ! If this value is -1 then no far-field FMM interactions are performed.
182 integer, intent(in) :: pl
183 ! Number of OpenMP threads to be used.
184 integer, intent(in) :: nproc
185 ! Number of atoms in the molecule.
186 integer, intent(in) :: nsph
187 ! Charges of atoms of a dimension (nsph).
188 real(dp), intent(in) :: charge(nsph)
189 ! Centers of atoms of a dimension (3, nsph).
190 real(dp), intent(in) :: csph(3, nsph)
191 ! Array of radii of atoms of a dimension (nsph).
192 real(dp), intent(in) :: rsph(nsph)
193 ! Error printing function
194 procedure(print_func_interface) :: print_func
195 ! log file name
196 character(len=255) :: output_filename
197 !! Outputs
198 type(ddx_params_type), intent(out) :: params
199 !! Local variables
200 integer :: igrid, i, info
201 character(len=255) :: string
202 !! The code
203 ! Clear error state
204 params % error_flag = 0
205 params % error_message = ''
206 ! parse the log file name
207 if (len(trim(output_filename)) .ne. 0) then
208 params % output_filename = output_filename
209 params % verbose = .true.
210 params % len_output_filename = len(trim(output_filename))
211 else
212 params % output_filename = ''
213 params % verbose = .false.
214 params % len_output_filename = 0
215 end if
216 ! Model, 1=COSMO, 2=PCM, 3=LPB
217 if ((model .lt. 1) .or. (model .gt. 3)) then
218 params % error_flag = 1
219 params % error_message = "params_init: invalid value of `model`"
220 return
221 end if
222 params % model = model
223 ! Check if forces are needed
224 if ((force .lt. 0) .or. (force .gt. 1)) then
225 params % error_flag = 1
226 params % error_message = "params_init: invalid value of `force`"
227 return
228 end if
229 params % force = force
230 ! Relative dielectric permittivity
231 if (eps .le. one) then
232 params % error_flag = 1
233 params % error_message = "params_init: invalid value of `eps`"
234 return
235 end if
236 params % eps = eps
237 ! Debye-H\"{u}ckel parameter (only used in ddLPB)
238 if ((model .eq. 3) .and. (kappa .le. zero)) then
239 params % error_flag = 1
240 params % error_message = "params_init: invalid value of `kappa`"
241 return
242 end if
243 params % kappa = kappa
244 ! Regularization parameter
245 if ((eta .lt. zero) .or. (eta .gt. one)) then
246 params % error_flag = 1
247 params % error_message = "params_init: invalid value of `eta`"
248 return
249 end if
250 params % eta = eta
251 ! Shift of a regularization
252 if ((se .lt. -one) .or. (se .gt. one)) then
253 params % error_flag = 1
254 params % error_message = "params_init: invalid value of `se`"
255 return
256 end if
257 params % se = se
258 ! Degree of modeling spherical harmonics
259 if (lmax .lt. 0) then
260 params % error_flag = 1
261 params % error_message = "params_init: invalid value of `lmax`"
262 return
263 end if
264 params % lmax = lmax
265 ! Check number of Lebedev grid points
266 igrid = 0
267 do i = 1, nllg
268 if (ng0(i) .eq. ngrid) then
269 igrid = i
270 exit
271 end if
272 end do
273 if (igrid .eq. 0) then
274 params % error_flag = 1
275 params % error_message = "params_init: Unsupported value of `ngrid`"
276 return
277 end if
278 params % ngrid = ngrid
279 ! Maximum number of iterations
280 if (maxiter .le. 0) then
281 params % error_flag = 1
282 params % error_message = "params_init: invalid value of `maxiter`"
283 return
284 end if
285 params % maxiter = maxiter
286 ! Number of Jacobi DIIS extrapolation points (ndiis=25 works)
287 if (jacobi_ndiis .lt. 0) then
288 params % error_flag = 1
289 params % error_message = "params_init: invalid value of `jacobi_ndiis`"
290 return
291 end if
292 params % jacobi_ndiis = jacobi_ndiis
293 ! Check if FMM-acceleration is needed
294 if ((fmm .lt. 0) .or. (fmm .gt. 1)) then
295 params % error_flag = 1
296 params % error_message = "params_init: invalid value of `fmm`"
297 return
298 end if
299 params % fmm = fmm
300 ! Set FMM parameters if FMM is needed
301 if (fmm .eq. 1) then
302 ! Maximal degree of multipole spherical harmonics. Value -1 means no
303 ! far-field interactions are to be computed, only near-field
304 ! interactions are taken into account.
305 if (pm .lt. -1) then
306 params % error_flag = 1
307 params % error_message = "params_init: invalid value of `pm`"
308 return
309 end if
310 ! Maximal degree of local spherical harmonics. Value -1 means no
311 ! far-field interactions are to be computed, only near-field
312 ! interactions are taken into account.
313 if (pl .lt. -1) then
314 params % error_flag = 1
315 params % error_message = "params_init: invalid value of `pl`"
316 return
317 end if
318 ! If far-field interactions are to be ignored
319 if ((pl .eq. -1) .or. (pm .eq. -1)) then
320 params % pm = -1
321 params % pl = -1
322 ! If far-field interactions are to be taken into account
323 else
324 params % pm = pm
325 params % pl = pl
326 end if
327 else
328 ! These values are ignored if fmm flag is 0
329 params % pm = -2
330 params % pl = -2
331 end if
332 ! Number of OpenMP threads to be used
333 ! available.
334 if (nproc .lt. 0) then
335 params % error_flag = 1
336 params % error_message = "params_init: invalid value of `nproc`"
337 return
338 else if (nproc .eq. 0) then
339 params % nproc = 1
340 else
341 params % nproc = nproc
342 end if
343 call omp_set_num_threads(params % nproc)
344 ! Number of atoms
345 if (nsph .le. 0) then
346 params % error_flag = 1
347 params % error_message = "params_init: invalid value of `nsph`"
348 return
349 end if
350 params % nsph = nsph
351 ! Allocation of charge, csph and rsph
352 allocate(params % charge(nsph), params % csph(3, nsph), &
353 & params % rsph(nsph), stat=info)
354 if (info .ne. 0) then
355 params % error_flag = 1
356 params % error_message = "params_init: `charge`, `csph` and `rsph` " &
357 & // "allocations failed"
358 return
359 end if
360 params % charge = charge
361 params % csph = csph
362 params % rsph = rsph
363 if (matvecmem.eq.0 .or. matvecmem.eq.1) then
364 params % matvecmem = matvecmem
365 else
366 params % error_flag = 1
367 params % error_message = "params_init: invalid value of `matvecmem`"
368 return
369 end if
370 ! Set print function for errors
371 params % print_func => print_func
372 ! init log
373 call init_printing(params)
374end subroutine params_init

473subroutine closest_supported_lebedev_grid(ngrid)
474 integer, intent(inout) :: ngrid
475 integer :: igrid, i, inear, jnear
476 ! Get nearest number of Lebedev grid points
477 igrid = 0
478 inear = 100000
479 do i = 1, nllg
480 jnear = abs(ng0(i) - ngrid)
481 if (jnear .lt. inear) then
482 inear = jnear
483 igrid = i
484 end if
485 end do
486 ngrid = ng0(igrid)
487end subroutine

38subroutine ddlpb(params, constants, workspace, state, phi_cav, e_cav, &
39 & psi, tol, esolv, hessianphi_cav, force)
40 implicit none
41 type(ddx_params_type), intent(in) :: params
42 type(ddx_constants_type), intent(inout) :: constants
43 type(ddx_workspace_type), intent(inout) :: workspace
44 type(ddx_state_type), intent(inout) :: state
45 real(dp), intent(in) :: phi_cav(constants % ncav), &
46 & e_cav(3, constants % ncav), &
47 & psi( constants % nbasis, params % nsph), tol
48 real(dp), intent(out) :: esolv
49 real(dp), intent(out), optional :: force(3, params % nsph)
50 real(dp), intent(in), optional :: hessianphi_cav(3, 3, constants % ncav)
51
52 call ddlpb_setup(params, constants, workspace, state, phi_cav, &
53 & e_cav, psi)
54 if (workspace % error_flag .eq. 1) return
55 call ddlpb_guess(params, constants, workspace, state, tol)
56 if (workspace % error_flag .eq. 1) return
57 call ddlpb_solve(params, constants, workspace, state, tol)
58 if (workspace % error_flag .eq. 1) return
59
60 ! Compute the solvation energy
61 call ddlpb_energy(constants, state, esolv)
62
63 ! Get forces if needed
64 if(params % force .eq. 1) then
65 call ddlpb_guess_adjoint(params, constants, workspace, state, tol)
66 if (workspace % error_flag .eq. 1) return
67 call ddlpb_solve_adjoint(params, constants, workspace, state, tol)
68 if (workspace % error_flag .eq. 1) return
69 call ddlpb_solvation_force_terms(params, constants, workspace, &
70 & state, hessianphi_cav, force)
71 if (workspace % error_flag .eq. 1) return
72 endif
73
74end subroutine ddlpb

88subroutine ddlpb_setup(params, constants, workspace, state, phi_cav, &
89 & e_cav, psi)
90 implicit none
91 type(ddx_params_type), intent(in) :: params
92 type(ddx_constants_type), intent(inout) :: constants
93 type(ddx_workspace_type), intent(inout) :: workspace
94 type(ddx_state_type), intent(inout) :: state
95 real(dp), intent(in) :: phi_cav(constants % ncav)
96 real(dp), intent(in) :: e_cav(3, constants % ncav)
97 real(dp), intent(in) :: psi(constants % nbasis, params % nsph)
98
99 state % psi = psi
100 state % rhs_adj_lpb(:, :, 1) = psi
101 ! in ddLPB the polarization density X is expanded according to a
102 ! slightly different definition which lacks the 4pi/(2l + 1) factors
103 ! which are present in ddCOSMO and ddPCM. This affect the adjoint
104 ! density as well. To have a consistent adjoint density with ddCOSMO
105 ! and ddPCM, we have to scale the RHS.
106 call convert_ddcosmo(params, constants, -1, state % rhs_adj_lpb(:,:,1))
107 state % rhs_adj_lpb(:, :, 2) = 0.0d0
108
109 !! Setting initial values to zero
110 state % g_lpb = zero
111 state % f_lpb = zero
112 state % phi_grid = zero
113
114 ! Unwrap sparsely stored potential at cavity points phi_cav into phi_grid
115 ! and multiply it by characteristic function at cavity points ui
116 call ddcav_to_grid_work(params % ngrid, params % nsph, &
117 & constants % ncav, constants % icav_ia, &
118 & constants % icav_ja, phi_cav, state % phi_grid)
119 workspace % tmp_cav = phi_cav * constants % ui_cav
120 call ddcav_to_grid_work(params % ngrid, params % nsph, &
121 & constants % ncav, constants % icav_ia, &
122 & constants % icav_ja, workspace % tmp_cav, &
123 & workspace % tmp_grid)
124 state % g_lpb = - workspace % tmp_grid
125
126 ! store the gradient of the potential (- electric field)
127 state % gradphi_cav = - e_cav
128
129 ! wghpot_f : Intermediate computation of F_0 Eq.(75) from QSM19.SISC
130 call wghpot_f(params, constants, workspace, state % gradphi_cav, state % f_lpb)
131
132 ! Setting of the local variables
133 state % rhs_lpb = zero
134
135 ! integrate RHS
136 call ddintegrate(params % nsph, constants % nbasis, &
137 & params % ngrid, constants % vwgrid, &
138 & constants % vgrid_nbasis, state % g_lpb, state % rhs_lpb(:,:,1))
139 call ddintegrate(params % nsph, constants % nbasis, &
140 & params % ngrid, constants % vwgrid, &
141 & constants % vgrid_nbasis, state % f_lpb, state % rhs_lpb(:,:,2))
142 state % rhs_lpb(:,:,1) = state % rhs_lpb(:,:,1) + state % rhs_lpb(:,:,2)
143
144end subroutine ddlpb_setup

226subroutine ddlpb_solve(params, constants, workspace, state, tol)
227 implicit none
228 type(ddx_params_type), intent(in) :: params
229 type(ddx_constants_type), intent(inout) :: constants
230 type(ddx_workspace_type), intent(inout) :: workspace
231 type(ddx_state_type), intent(inout) :: state
232 real(dp), intent(in) :: tol
233 real(dp) :: start_time
234
235 state % x_lpb_niter = params % maxiter
236
237 ! solve LS using Jacobi/DIIS
238 start_time = omp_get_wtime()
239 call jacobi_diis_external(params, constants, workspace, &
240 & 2*constants % n, tol, state % rhs_lpb, state % x_lpb, &
241 & state % x_lpb_niter, state % x_lpb_rel_diff, cx, prec_tx, rmsnorm)
242 if (workspace % error_flag .ne. 0) then
243 workspace % error_message = 'Jacobi solver failed to converge " // &
244 & "in ddlpb_solve'
245 return
246 end if
247 state % x_lpb_time = omp_get_wtime() - start_time
248 ! in ddLPB the polarization density X is expanded according to a
249 ! slightly different definition which lacks the 4pi/(2l + 1) factors
250 ! which are present in ddCOSMO and ddPCM. Before exiting, we scale
251 ! the density so that it is consistent with the ddCOSMO and ddPCM
252 ! ones.
253 call convert_ddcosmo(params, constants, -1, state % x_lpb(:,:,1))
254end subroutine ddlpb_solve

309subroutine ddlpb_solvation_force_terms(params, constants, workspace, &
310 & state, hessianphi_cav, force)
311 implicit none
312 type(ddx_params_type), intent(in) :: params
313 type(ddx_constants_type), intent(in) :: constants
314 type(ddx_workspace_type), intent(inout) :: workspace
315 type(ddx_state_type), intent(inout) :: state
316 real(dp), intent(in) :: hessianphi_cav(3, 3, constants % ncav)
317 real(dp), intent(out) :: force(3, params % nsph)
318
319 ! local
320 real(dp), dimension(constants % nbasis) :: basloc, vplm
321 real(dp), dimension(3, constants % nbasis) :: dbasloc
322 real(dp), dimension(params % lmax + 1) :: vsin, vcos
323
324 ! large local are allocatable
325 real(dp), allocatable :: ef(:,:), normal_hessian_cav(:,:), &
326 & diff_re(:,:), scaled_xr(:,:)
327 integer :: isph, icav, igrid, istat
328 integer :: i
329 real(dp), external :: ddot, dnrm2
330 real(dp) :: start_time, finish_time
331
332 start_time = omp_get_wtime()
333 allocate(ef(3, params % nsph), &
334 & normal_hessian_cav(3, constants % ncav), &
335 & diff_re(constants % nbasis, params % nsph), &
336 & scaled_xr(constants % nbasis, params % nsph), stat=istat)
337 if (istat.ne.0) then
338 workspace % error_message = 'Allocation failed in ddlpb_force_worker'
339 workspace % error_flag = 1
340 return
341 end if
342
343 ! we have to insert again the factor that we removed in ddlpb_solve
344 call convert_ddcosmo(params, constants, 1, state % x_lpb(:, :, 1))
345
346 diff_re = zero
347 vsin = zero
348 vcos = zero
349 vplm = zero
350 basloc = zero
351 dbasloc = zero
352 ef = zero
353 force = zero
354
355 ! Compute the derivative of the normal derivative of psi_0
356 normal_hessian_cav = zero
357 icav = 0
358 do isph = 1, params % nsph
359 do igrid = 1, params % ngrid
360 if(constants % ui(igrid, isph) .gt. zero) then
361 icav = icav + 1
362 do i = 1, 3
363 normal_hessian_cav(:, icav) = normal_hessian_cav(:,icav) +&
364 & hessianphi_cav(:,i,icav)*constants % cgrid(i,igrid)
365 end do
366 end if
367 end do
368 end do
369
370 ! Scale by the factor of 1/(4Pi/(2l+1))
371 scaled_xr = state % x_lpb(:,:,1)
372 call convert_ddcosmo(params, constants, -1, scaled_xr)
373
374 !$omp parallel do default(none) shared(params,constants,workspace, &
375 !$omp scaled_xr,state,force) private(isph,basloc,dbasloc,vplm,vcos,vsin) &
376 !$omp schedule(static,1)
377 do isph = 1, params % nsph
378 ! Compute A^k*Xadj_r, using Subroutine from ddCOSMO
379 call contract_grad_l(params, constants, isph, scaled_xr, &
380 & state % x_adj_r_grid, basloc, dbasloc, vplm, vcos, vsin, &
381 & force(:,isph))
382 ! Compute B^k*Xadj_e
383 call contract_grad_b(params, constants, isph, state % x_lpb(:,:,2), &
384 & state % x_adj_e_grid, force(:, isph))
385 ! Computation of G0
386 call contract_grad_u(params, constants, isph, state % x_adj_r_grid, &
387 & state % phi_grid, force(:, isph))
388 end do
389 ! Compute C1 and C2 contributions
390 diff_re = zero
391 call contract_grad_c(params, constants, workspace, state % x_lpb(:,:,1), &
392 & state % x_lpb(:,:,2), state % x_adj_r_grid, state % x_adj_e_grid, &
393 & state % x_adj_lpb(:,:,1), state % x_adj_lpb(:,:,2), force, &
394 & diff_re)
395 ! Computation of G0 continued
396
397 ! NOTE: contract_grad_U returns a positive summation
398 if (workspace % error_flag .eq. 1) return
399 force = -force
400 icav = 0
401 do isph = 1, params % nsph
402 do igrid = 1, params % ngrid
403 if(constants % ui(igrid, isph) .ne. zero) then
404 icav = icav + 1
405 state % zeta(icav) = constants % wgrid(igrid) * &
406 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
407 & constants % vgrid(1, igrid), 1, &
408 & state % x_adj_lpb(1, isph, 1), 1)
409 end if
410 end do
411 end do
412
413 ! Computation of F0
414 call contract_grad_f(params, constants, workspace, &
415 & state % x_adj_lpb(:,:,1) + state % x_adj_lpb(:,:,2), &
416 & state % x_adj_re_grid, state % gradphi_cav, &
417 & normal_hessian_cav, force, state)
418 if (workspace % error_flag .eq. 1) return
419
420 force = pt5*force
421
422 deallocate(ef, normal_hessian_cav, diff_re, scaled_xr, stat=istat)
423 if (istat.ne.0) then
424 workspace % error_message = 'Deallocation failed in ddlpb_force_worker'
425 workspace % error_flag = 1
426 return
427 end if
428 finish_time = omp_get_wtime()
429 state % force_time = finish_time - start_time
430
431 ! and now we remove again the factor, so that the density is
432 ! restored.
433 call convert_ddcosmo(params, constants, -1, state % x_lpb(:, :, 1))
434
435end subroutine ddlpb_solvation_force_terms

445subroutine ddlpb_derivative_setup(params, constants, workspace, state)
446 implicit none
447 type(ddx_params_type), intent(in) :: params
448 type(ddx_constants_type), intent(in) :: constants
449 type(ddx_workspace_type), intent(inout) :: workspace
450 type(ddx_state_type), intent(inout) :: state
451
452 real(dp), allocatable :: coefy_d(:, :, :)
453 integer :: istat, jsph, igrid0, icav, isph, igrid, l0, m0, ind0, &
454 & inode, indl
455 real(dp) :: rijn, sum_int, fac
456 real(dp) :: work(constants % lmax0 + 1), coef(constants % nbasis0), &
457 & vij(3), vtij(3), sij(3)
458 complex(dp) :: work_complex(constants % lmax0 + 1)
459
460 ! expand the adjoint solutions at the Lebedev points and compute
461 ! their sum
462 call dgemm('T', 'N', params % ngrid, params % nsph, &
463 & constants % nbasis, one, constants % vgrid, &
464 & constants % vgrid_nbasis, state % x_adj_lpb(:, :, 1), &
465 & constants % nbasis, zero, state % x_adj_r_grid, params % ngrid)
466 call dgemm('T', 'N', params % ngrid, params % nsph, &
467 & constants % nbasis, one, constants % vgrid, &
468 & constants % vgrid_nbasis, state % x_adj_lpb(:, :, 2), &
469 & constants % nbasis, zero, state % x_adj_e_grid, params % ngrid)
470 state % x_adj_re_grid = state % x_adj_r_grid + state % x_adj_e_grid
471
472 if (params % fmm .eq. 0) then
473 allocate(coefy_d(constants % ncav, params % ngrid, params % nsph), &
474 & stat=istat)
475 if (istat.ne.0) then
476 workspace % error_flag = 1
477 workspace % error_message = &
478 & "allocation error in ddlpb_derivative_intermediates"
479 return
480 end if
481 coefy_d = zero
482 do jsph = 1, params % nsph
483 do igrid0 = 1, params % ngrid
484 icav = zero
485 do isph = 1, params % nsph
486 do igrid = 1, params % ngrid
487 if(constants % ui(igrid, isph) .gt. zero) then
488 icav = icav + 1
489 vij = params % csph(:,isph) + params % rsph(isph) &
490 & *constants % cgrid(:,igrid) &
491 & - params % csph(:,jsph)
492 rijn = sqrt(dot_product(vij,vij))
493 sij = vij/rijn
494 do l0 = 0, constants % lmax0
495 do m0 = -l0, l0
496 ind0 = l0**2 + l0 + m0 + 1
497 coef(ind0) = &
498 & constants % vgrid(ind0, igrid0) * &
499 & constants % C_ik(l0, jsph)
500 end do
501 end do
502 vtij = vij*params % kappa
503 call fmm_m2p_bessel_work(vtij, constants % lmax0, &
504 & constants % vscales, &
505 & constants % sk_ri(:, jsph), one, &
506 & coef, zero, coefy_d(icav, igrid0, jsph), &
507 & work_complex, work)
508 end if
509 end do
510 end do
511 end do
512 end do
513 do jsph = 1, params % nsph
514 do igrid0 = 1, params % ngrid
515 icav = zero
516 sum_int = zero
517 do isph = 1, params % nsph
518 do igrid = 1, params % ngrid
519 if(constants % ui(igrid, isph) .gt. zero) then
520 icav = icav + 1
521 sum_int = sum_int + coefy_d(icav, igrid0, jsph) &
522 & *state % x_adj_re_grid(igrid, isph) &
523 & *constants % wgrid(igrid) &
524 & *constants % ui(igrid, isph)
525 end if
526 end do
527 end do
528 state % phi_n(igrid0, jsph) = &
529 & - (params % epsp/params % eps)*sum_int
530 end do
531 end do
532 else
533 ! Adjoint integration from spherical harmonics to grid points is not needed
534 ! here as ygrid already contains grid values, we just need to scale it by
535 ! weights of grid points
536 do isph = 1, params % nsph
537 workspace % tmp_grid(:, isph) = state % x_adj_re_grid(:, isph) * &
538 & constants % wgrid(:) * constants % ui(:, isph)
539 end do
540 ! Adjoint FMM
541 call tree_m2p_bessel_adj(params, constants, constants % lmax0, one, &
542 & workspace % tmp_grid, zero, params % lmax, workspace % tmp_sph)
543 call tree_l2p_bessel_adj(params, constants, one, &
544 & workspace % tmp_grid, zero, workspace % tmp_node_l)
545 call tree_l2l_bessel_rotation_adj(params, constants, &
546 & workspace % tmp_node_l)
547 call tree_m2l_bessel_rotation_adj(params, constants, &
548 & workspace % tmp_node_l, workspace % tmp_node_m)
549 call tree_m2m_bessel_rotation_adj(params, constants, &
550 & workspace % tmp_node_m)
551 ! Properly load adjoint multipole harmonics into tmp_sph
552 if(constants % lmax0 .lt. params % pm) then
553 do isph = 1, params % nsph
554 inode = constants % snode(isph)
555 workspace % tmp_sph(1:constants % nbasis0, isph) = &
556 & workspace % tmp_sph(1:constants % nbasis0, isph) + &
557 & workspace % tmp_node_m(1:constants % nbasis0, inode)
558 end do
559 else
560 indl = (params % pm+1)**2
561 do isph = 1, params % nsph
562 inode = constants % snode(isph)
563 workspace % tmp_sph(1:indl, isph) = &
564 & workspace % tmp_sph(1:indl, isph) + &
565 & workspace % tmp_node_m(:, inode)
566 end do
567 end if
568 ! Scale by C_ik
569 do isph = 1, params % nsph
570 do l0 = 0, constants % lmax0
571 ind0 = l0*l0 + l0 + 1
572 workspace % tmp_sph(ind0-l0:ind0+l0, isph) = &
573 & workspace % tmp_sph(ind0-l0:ind0+l0, isph) * &
574 & constants % C_ik(l0, isph)
575 end do
576 end do
577 ! Multiply by vgrid
578 call dgemm('T', 'N', params % ngrid, params % nsph, &
579 & constants % nbasis0, -params % epsp/params % eps, &
580 & constants % vgrid, constants % vgrid_nbasis, &
581 & workspace % tmp_sph, constants % nbasis, zero, &
582 & state % phi_n, params % ngrid)
583 end if
584
585 ! assemble the pseudo-dipoles
586 icav = 0
587 do isph = 1, params % nsph
588 do igrid = 1, params % ngrid
589 if (constants % ui(igrid, isph) .gt. zero) then
590 icav = icav + 1
591 fac = - constants % ui(igrid, isph)*constants % wgrid(igrid) &
592 & *state % phi_n(igrid, isph)
593 state % zeta_dip(1, icav) = fac*constants % cgrid(1, igrid)
594 state % zeta_dip(2, icav) = fac*constants % cgrid(2, igrid)
595 state % zeta_dip(3, icav) = fac*constants % cgrid(3, igrid)
596 end if
597 end do
598 end do
599
600 if (allocated(coefy_d)) then
601 deallocate(coefy_d, stat=istat)
602 if (istat.ne.0) then
603 workspace % error_flag = 1
604 workspace % error_message = &
605 & "deallocation error in ddlpb_derivative_intermediates"
606 return
607 end if
608 end if
609
610end subroutine ddlpb_derivative_setup

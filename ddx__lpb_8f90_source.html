152!! A standalone ddLPB calculation happens here

160subroutine ddlpb(params, constants, workspace, state, phi_cav, gradphi_cav, &
161 & hessianphi_cav, psi, tol, esolv, force)
162 implicit none
163 type(ddx_params_type), intent(in) :: params
164 type(ddx_constants_type), intent(inout) :: constants
165 type(ddx_workspace_type), intent(inout) :: workspace
166 type(ddx_state_type), intent(inout) :: state
167 real(dp), intent(in) :: phi_cav(constants % ncav), &
168 & gradphi_cav(3, constants % ncav), &
169 & hessianphi_cav(3, 3, constants % ncav), &
170 & psi( constants % nbasis, params % nsph), tol
171 real(dp), intent(out) :: esolv, force(3, params % nsph)
172 real(dp), external :: ddot
173
174 ! TODO: find a consistent way to do the guess
175
176 call ddlpb_solve(params, constants, workspace, state, phi_cav, &
177 & gradphi_cav, tol)
178 if (workspace % error_flag .eq. 1) return
179
180 ! Compute the solvation energy
181 ! note that psi is divided by fourpi
182 esolv = pt5*ddot(constants % n, state % x_lpb(:,:,1), 1, psi, 1) &
183 & /fourpi
184
185 ! Get forces if needed
186 if(params % force .eq. 1) then
187 call ddlpb_adjoint(params, constants, workspace, state, psi, tol)
188 if (workspace % error_flag .eq. 1) return
189 call ddlpb_force(params, constants, workspace, state, phi_cav, &
190 & gradphi_cav, hessianphi_cav, psi, force)
191 if (workspace % error_flag .eq. 1) return
192 endif

196subroutine ddlpb_solve_worker(params, constants, workspace, phi_cav, &
197 & gradphi_cav, g_lpb, f_lpb, phi_grid, x_lpb, x_lpb_niter, &
198 & x_lpb_time, x_lpb_rel_diff, tol)
199 implicit none
200 type(ddx_params_type), intent(in) :: params
201 type(ddx_constants_type), intent(inout) :: constants
202 type(ddx_workspace_type), intent(inout) :: workspace
203 real(dp), intent(in) :: phi_cav(constants % ncav)
204 real(dp), intent(in) :: gradphi_cav(3, constants % ncav)
205 real(dp), intent(inout) :: x_lpb(constants % nbasis, params % nsph, 2)
206 integer, intent(inout) :: x_lpb_niter
207 real(dp), intent(out) :: x_lpb_time
208 real(dp), intent(out) :: x_lpb_rel_diff(x_lpb_niter)
209 real(dp), intent(in) :: tol
210 real(dp), intent(out) :: g_lpb(params % ngrid, params % nsph)
211 real(dp), intent(out) :: f_lpb(params % ngrid, params % nsph)
212 real(dp), intent(out) :: phi_grid(params % ngrid, params % nsph)
213
214 integer :: istat
215 real(dp) :: start_time
216
217 real(dp), allocatable :: rhs(:,:,:)
218
219 allocate(rhs(constants % nbasis, params % nsph, 2), stat=istat)
220 if (istat.ne.0) then
221 workspace % error_message = 'Allocation failed in ddlpb_solve_worker'
222 workspace % error_flag = 1
223 return
224 end if
225
226 !! Use a tighter tolerance for the microiterations to ensure convergence
227 constants % inner_tol = tol/100.0d0
228
229 !! Setting initial values to zero
230 g_lpb = zero
231 f_lpb = zero
232 phi_grid = zero
233
234 ! Unwrap sparsely stored potential at cavity points phi_cav into phi_grid
235 ! and multiply it by characteristic function at cavity points ui
236 call ddcav_to_grid_work(params % ngrid, params % nsph, &
237 & constants % ncav, constants % icav_ia, &
238 & constants % icav_ja, phi_cav, phi_grid)
239 workspace % tmp_cav = phi_cav * constants % ui_cav
240 call ddcav_to_grid_work(params % ngrid, params % nsph, &
241 & constants % ncav, constants % icav_ia, &
242 & constants % icav_ja, workspace % tmp_cav, &
243 & workspace % tmp_grid)
244 g_lpb = - workspace % tmp_grid
245
246 ! wghpot_f : Intermediate computation of F_0 Eq.(75) from QSM19.SISC
247 call wghpot_f(params, constants, workspace, gradphi_cav, f_lpb)
248
249 ! Setting of the local variables
250 rhs = zero
251
252 ! integrate RHS
253 call ddintegrate(params % nsph, constants % nbasis, &
254 & params % ngrid, constants % vwgrid, &
255 & constants % vgrid_nbasis, g_lpb, rhs(:,:,1))
256 call ddintegrate(params % nsph, constants % nbasis, &
257 & params % ngrid, constants % vwgrid, &
258 & constants % vgrid_nbasis, f_lpb, rhs(:,:,2))
259 rhs(:,:,1) = rhs(:,:,1) + rhs(:,:,2)
260
261 ! guess
262 workspace % ddcosmo_guess = zero
263 workspace % hsp_guess = zero
264 call prec_tx(params, constants, workspace, rhs, x_lpb)
265
266 ! solve LS using Jacobi/DIIS
267 start_time = omp_get_wtime()
268 call jacobi_diis_external(params, constants, workspace, &
269 & 2*constants % n, tol, rhs, x_lpb, x_lpb_niter, x_lpb_rel_diff, &
270 & cx, prec_tx, rmsnorm)
271 if (workspace % error_flag .ne. 0) then
272 workspace % error_message = 'Jacobi solver failed to converge in ddlpb_solve_worker'
273 return
274 end if
275 x_lpb_time = omp_get_wtime() - start_time
276
277 deallocate(rhs, stat = istat)
278 if (istat .ne. 0) then
279 workspace % error_message = 'Deallocation failed in ddlpb_solve_worker'
280 workspace % error_flag = 1
281 return
282 end if
283end subroutine ddlpb_solve_worker

285subroutine ddlpb_adjoint_worker(params, constants, workspace, psi, tol, &
286 & x_adj_lpb, x_adj_lpb_niter, x_adj_lpb_time, x_adj_lpb_rel_diff)
287 implicit none
288 type(ddx_params_type), intent(in) :: params
289 type(ddx_constants_type), intent(inout) :: constants
290 type(ddx_workspace_type), intent(inout) :: workspace
291 real(dp), intent(in) :: psi(constants % nbasis, params % nsph)
292 real(dp), intent(in) :: tol
293 real(dp), intent(out) :: x_adj_lpb(constants % nbasis, params % nsph, 2)
294 integer, intent(inout) :: x_adj_lpb_niter
295 real(dp), intent(out) :: x_adj_lpb_time
296 real(dp), intent(out) :: x_adj_lpb_rel_diff(x_adj_lpb_niter)
297
298 real(dp), allocatable :: rhs(:,:,:)
299 real(dp) :: start_time
300 integer :: istat
301
302 allocate(rhs(constants % nbasis, params % nsph, 2), stat=istat)
303 if (istat.ne.0) then
304 workspace % error_message = 'Allocation failed in ddlpb_adjoint_worker'
305 workspace % error_flag = 1
306 return
307 end if
308
309 !! Use a tighter tolerance for the microiterations to ensure convergence
310 constants % inner_tol = tol/100.0d0
311
312 ! Psi shall be divided by a factor 4pi for the LPB case
313 ! It is intended to take into account this constant in the LPB
314
315 ! set up the RHS
316 rhs(:,:,1) = psi/fourpi
317 rhs(:,:,2) = zero
318
319 ! guess
320 workspace % ddcosmo_guess = zero
321 workspace % hsp_guess = zero
322 call prec_tstarx(params, constants, workspace, rhs, x_adj_lpb)
323
324 ! solve adjoint LS using Jacobi/DIIS
325 start_time = omp_get_wtime()
326 call jacobi_diis_external(params, constants, workspace, &
327 & 2*constants % n, tol, rhs, x_adj_lpb, x_adj_lpb_niter, &
328 & x_adj_lpb_rel_diff, cstarx, prec_tstarx, &
329 & rmsnorm)
330 if (workspace % error_flag .ne. 0) then
331 workspace % error_message = 'Jacobi solver failed to converge in ddlpb_adjoint_worker'
332 return
333 end if
334 x_adj_lpb_time = omp_get_wtime() - start_time
335 deallocate(rhs, stat=istat)
336 if (istat.ne.0) then
337 workspace % error_message = 'Deallocation failed in ddlpb_adjoint_worker'
338 workspace % error_flag = 1
339 return
340 end if
341end subroutine ddlpb_adjoint_worker

343subroutine ddlpb_force_worker(params, constants, workspace, hessian, &
344 & phi_grid, gradphi, x, x_adj, zeta, force)
345 type(ddx_params_type), intent(in) :: params
346 type(ddx_constants_type), intent(in) :: constants
347 type(ddx_workspace_type), intent(inout) :: workspace
348
349 real(dp), dimension(3, 3, constants % ncav), intent(in) :: hessian
350 real(dp), dimension(params % ngrid, params % nsph), intent(in) :: phi_grid
351 real(dp), dimension(3, constants % ncav), intent(in) :: gradphi
352 real(dp), dimension(constants % nbasis, params % nsph, 2), intent(in) :: x, x_adj
353 real(dp), dimension(3, params % nsph), intent(out) :: force
354 real(dp), intent(out) :: zeta(constants % ncav)
355
356 ! local
357 real(dp), dimension(constants % nbasis) :: basloc, vplm
358 real(dp), dimension(3, constants % nbasis) :: dbasloc
359 real(dp), dimension(params % lmax + 1) :: vsin, vcos
360
361 ! large local are allocatable
362 real(dp), allocatable :: ef(:,:), xadj_r_sgrid(:,:), xadj_e_sgrid(:,:), &
363 & normal_hessian_cav(:,:), diff_re(:,:), scaled_xr(:,:)
364 integer :: isph, icav, icav_gr, icav_ge, igrid, istat
365 integer :: i, inode, jnear, inear, jnode, jsph
366 real(dp), external :: ddot, dnrm2
367 real(dp) :: tcontract_gradi_Lik, tcontract_gradi_Lji, &
368 & tcontract_gradi_Bik, tcontract_gradi_Bji, tcontract_grad_U, &
369 & tcontract_grad_C_worker1, tcontract_grad_C_worker2
370 real(dp) :: d(3), dnorm, tmp1, tmp2
371
372 allocate(ef(3, params % nsph), &
373 & xadj_r_sgrid(params % ngrid, params % nsph), &
374 & xadj_e_sgrid(params % ngrid, params % nsph), &
375 & normal_hessian_cav(3, constants % ncav), &
376 & diff_re(constants % nbasis, params % nsph), &
377 & scaled_xr(constants % nbasis, params % nsph), stat=istat)
378 if (istat.ne.0) then
379 workspace % error_message = 'Allocation failed in ddlpb_force_worker'
380 workspace % error_flag = 1
381 return
382 end if
383
384 diff_re = zero
385 vsin = zero
386 vcos = zero
387 vplm = zero
388 basloc = zero
389 dbasloc = zero
390 ef = zero
391 force = zero
392
393 ! Compute the derivative of the normal derivative of psi_0
394 normal_hessian_cav = zero
395 icav = 0
396 do isph = 1, params % nsph
397 do igrid = 1, params % ngrid
398 if(constants % ui(igrid, isph) .gt. zero) then
399 icav = icav + 1
400 do i = 1, 3
401 normal_hessian_cav(:, icav) = normal_hessian_cav(:,icav) +&
402 & hessian(:,i,icav)*constants % cgrid(i,igrid)
403 end do
404 end if
405 end do
406 end do
407
408 ! Call dgemm to integrate the adjoint solution on the grid points
409 call dgemm('T', 'N', params % ngrid, params % nsph, constants % nbasis, &
410 & one, constants % vgrid, constants % vgrid_nbasis, x_adj(:,:,1), &
411 & constants % nbasis, zero, xadj_r_sgrid, params % ngrid)
412 call dgemm('T', 'N', params % ngrid, params % nsph, constants % nbasis, &
413 & one, constants % vgrid, constants % vgrid_nbasis, x_adj(:,:,2), &
414 & constants % nbasis, zero, xadj_e_sgrid, params % ngrid)
415
416 ! Scale by the factor of 1/(4Pi/(2l+1))
417 scaled_xr = x(:,:,1)
418 call convert_ddcosmo(params, constants, -1, scaled_xr)
419
420 !$omp parallel do default(none) shared(params,constants,workspace, &
421 !$omp scaled_xr,xadj_r_sgrid,force,x,xadj_e_sgrid,phi_grid) private(isph, &
422 !$omp basloc,dbasloc,vplm,vcos,vsin)
423 do isph = 1, params % nsph
424 ! Compute A^k*Xadj_r, using Subroutine from ddCOSMO
425 call contract_grad_l(params, constants, isph, scaled_xr, xadj_r_sgrid, &
426 & basloc, dbasloc, vplm, vcos, vsin, force(:,isph))
427 ! Compute B^k*Xadj_e
428 call contract_grad_b(params, constants, workspace, isph, x(:,:,2), xadj_e_sgrid, &
429 & basloc, dbasloc, vplm, vcos, vsin, force(:, isph))
430 ! Computation of G0
431 call contract_grad_u(params, constants, isph, xadj_r_sgrid, phi_grid, &
432 & force(:, isph))
433 end do
434 ! Compute C1 and C2 contributions
435 diff_re = zero
436 call contract_grad_c(params, constants, workspace, x(:,:,1), x(:,:,2), xadj_r_sgrid, &
437 & xadj_e_sgrid, x_adj(:,:,1), x_adj(:,:,2), force, diff_re)
438 ! Computation of G0 continued
439 ! NOTE: contract_grad_U returns a positive summation
440 if (workspace % error_flag .eq. 1) return
441 force = -force
442 icav = 0
443 do isph = 1, params % nsph
444 do igrid = 1, params % ngrid
445 if(constants % ui(igrid, isph) .ne. zero) then
446 icav = icav + 1
447 zeta(icav) = -constants % wgrid(igrid) * &
448 & constants % ui(igrid, isph) * ddot(constants % nbasis, &
449 & constants % vgrid(1, igrid), 1, &
450 & x_adj(1, isph, 1), 1)
451 force(:, isph) = force(:, isph) + &
452 & zeta(icav)*gradphi(:, icav)
453 end if
454 end do
455 end do
456 !! Last term where we compute gradients of potential at centers of atoms
457 !! spawned by intermediate zeta.
458 if(params % fmm .eq. 1) then
459 !! This step can be substituted by a proper dgemm if zeta
460 !! intermediate is converted from cavity points to all grid points
461 !! with zero values at internal grid points
462 ! P2M step
463 icav = 0
464 do isph = 1, params % nsph
465 inode = constants % snode(isph)
466 workspace % tmp_node_m(:, inode) = zero
467 do igrid = 1, params % ngrid
468 if(constants % ui(igrid, isph) .eq. zero) cycle
469 icav = icav + 1
470 call fmm_p2m(constants % cgrid(:, igrid), zeta(icav), &
471 & one, params % pm, constants % vscales, one, &
472 & workspace % tmp_node_m(:, inode))
473 end do
474 workspace % tmp_node_m(:, inode) = workspace % tmp_node_m(:, inode) / &
475 & params % rsph(isph)
476 end do
477 ! M2M, M2L and L2L translations
478 call tree_m2m_rotation(params, constants, workspace % tmp_node_m)
479 call tree_m2l_rotation(params, constants, workspace % tmp_node_m, &
480 & workspace % tmp_node_l)
481 call tree_l2l_rotation(params, constants, workspace % tmp_node_l)
482 ! Now compute near-field FMM gradients
483 ! Cycle over all spheres
484 icav = 0
485 workspace % tmp_efld = zero
486 do isph = 1, params % nsph
487 ! Cycle over all external grid points
488 do igrid = 1, params % ngrid
489 if(constants % ui(igrid, isph) .eq. zero) cycle
490 icav = icav + 1
491 ! Cycle over all near-field admissible pairs of spheres,
492 ! including pair (isph, isph) which is a self-interaction
493 inode = constants % snode(isph)
494 do jnear = constants % snear(inode), constants % snear(inode+1)-1
495 ! Near-field interactions are possible only between leaf
496 ! nodes, which must contain only a single input sphere
497 jnode = constants % near(jnear)
498 jsph = constants % order(constants % cluster(1, jnode))
499 d = params % csph(:, isph) + &
500 & constants % cgrid(:, igrid)*params % rsph(isph) - &
501 & params % csph(:, jsph)
502 dnorm = dnrm2(3, d, 1)
503 workspace % tmp_efld(:, jsph) = workspace % tmp_efld(:, jsph) + &
504 & zeta(icav)*d/(dnorm**3)
505 end do
506 end do
507 end do
508 ! Take into account far-field FMM gradients only if pl > 0
509 if (params % pl .gt. 0) then
510 tmp1 = one / sqrt(3d0) / constants % vscales(1)
511 do isph = 1, params % nsph
512 inode = constants % snode(isph)
513 tmp2 = tmp1 / params % rsph(isph)
514 workspace % tmp_efld(3, isph) = workspace % tmp_efld(3, isph) + &
515 & tmp2*workspace % tmp_node_l(3, inode)
516 workspace % tmp_efld(1, isph) = workspace % tmp_efld(1, isph) + &
517 & tmp2*workspace % tmp_node_l(4, inode)
518 workspace % tmp_efld(2, isph) = workspace % tmp_efld(2, isph) + &
519 & tmp2*workspace % tmp_node_l(2, inode)
520 end do
521 end if
522 do isph = 1, params % nsph
523 force(:, isph) = force(:, isph) + &
524 & workspace % tmp_efld(:, isph)*params % charge(isph)
525 end do
526 ! Naive quadratically scaling implementation
527 else
528 ! This routines actually computes -grad, not grad
529 call efld(constants % ncav, zeta, constants % ccav, params % nsph, &
530 & params % csph, workspace % tmp_efld)
531 do isph = 1, params % nsph
532 force(:, isph) = force(:, isph) - &
533 & workspace % tmp_efld(:, isph)*params % charge(isph)
534 end do
535 end if
536
537 icav_gr = zero
538 icav_ge = zero
539 ! Computation of F0
540 call contract_grad_f(params, constants, workspace, x_adj(:,:,1), xadj_r_sgrid, &
541 & gradphi, normal_hessian_cav, icav_gr, force)
542 if (workspace % error_flag .eq. 1) return
543 call contract_grad_f(params, constants, workspace, x_adj(:,:,2), xadj_e_sgrid, &
544 & gradphi, normal_hessian_cav, icav_ge, force)
545 if (workspace % error_flag .eq. 1) return
546
547 force = - pt5*force
548
549 deallocate(ef, xadj_r_sgrid, xadj_e_sgrid, normal_hessian_cav, &
550 & diff_re, scaled_xr, stat=istat)

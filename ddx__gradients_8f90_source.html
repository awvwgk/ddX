36subroutine contract_gradi_lik(params, constants, isph, sigma, xi, basloc, dbsloc, vplm, vcos, vsin, fx )
37 type(ddx_params_type), intent(in) :: params
38 type(ddx_constants_type), intent(in) :: constants
39 integer, intent(in) :: isph
40 real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: sigma
41 real(dp), dimension(params % ngrid), intent(in) :: xi
42 real(dp), dimension(constants % nbasis), intent(inout) :: basloc, vplm
43 real(dp), dimension(3, constants % nbasis), intent(inout) :: dbsloc
44 real(dp), dimension(params % lmax+1), intent(inout) :: vcos, vsin
45 real(dp), dimension(3), intent(inout) :: fx
46 integer :: ig, ij, jsph, l, ind, m
47 real(dp) :: vvij, tij, xij, oij, t, fac, fl, f1, f2, f3, beta, tlow, thigh
48 real(dp) :: vij(3), sij(3), alp(3), va(3)
49 real(dp), external :: dnrm2
50 tlow = one - pt5*(one - params % se)*params % eta
51 thigh = one + pt5*(one + params % se)*params % eta
52
53 do ig = 1, params % ngrid
54 va = zero
55 do ij = constants % inl(isph), constants % inl(isph+1) - 1
56 jsph = constants % nl(ij)
57 vij = params % csph(:,isph) + &
58 & params % rsph(isph)*constants % cgrid(:,ig) - &
59 & params % csph(:,jsph)
60 !vvij = sqrt(dot_product(vij,vij))
61 vvij = dnrm2(3, vij, 1)
62 tij = vvij/params % rsph(jsph)
63
64 if (tij.ge.thigh) cycle
65
66 sij = vij/vvij
67 !call dbasis(sij,basloc,dbsloc,vplm,vcos,vsin)
68 call dbasis(params, constants, sij, basloc, dbsloc, vplm, vcos, vsin)
69 alp = zero
70 t = one
71 do l = 1, params % lmax
72 ind = l*l + l + 1
73 fl = dble(l)
74 fac = t/(constants % vscales(ind)**2)
75 do m = -l, l
76 f2 = fac*sigma(ind+m,jsph)
77 f1 = f2*fl*basloc(ind+m)
78 alp(:) = alp(:) + f1*sij(:) + f2*dbsloc(:,ind+m)
79 end do
80 t = t*tij
81 end do
82 beta = intmlp(params, constants, tij,sigma(:,jsph),basloc)
83 xij = fsw(tij, params % se, params % eta)
84 if (constants % fi(ig,isph).gt.one) then
85 oij = xij/constants % fi(ig,isph)
86 f2 = -oij/constants % fi(ig,isph)
87 else
88 oij = xij
89 f2 = zero
90 end if
91 f1 = oij/params % rsph(jsph)
92 va(:) = va(:) + f1*alp(:) + beta*f2*constants % zi(:,ig,isph)
93 if (tij .gt. tlow) then
94 f3 = beta*dfsw(tij,params % se,params % eta)/params % rsph(jsph)
95 if (constants % fi(ig,isph).gt.one) f3 = f3/constants % fi(ig,isph)
96 va(:) = va(:) + f3*sij(:)
97 end if
98 end do
99 fx = fx - constants % wgrid(ig)*xi(ig)*va(:)
100 end do
101end subroutine contract_gradi_lik

104subroutine contract_gradi_lji(params, constants, isph, sigma, xi, basloc, dbsloc, vplm, vcos, vsin, fx )
105 type(ddx_params_type), intent(in) :: params
106 type(ddx_constants_type), intent(in) :: constants
107 integer, intent(in) :: isph
108 real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: sigma
109 real(dp), dimension(params % ngrid, params % nsph), intent(in) :: xi
110 real(dp), dimension(constants % nbasis), intent(inout) :: basloc, vplm
111 real(dp), dimension(3, constants % nbasis), intent(inout) :: dbsloc
112 real(dp), dimension(params % lmax+1), intent(inout) :: vcos, vsin
113 real(dp), dimension(3), intent(inout) :: fx
114
115 integer :: ig, ji, jsph, l, ind, m, jk, ksph
116 logical :: proc
117 real(dp) :: vvji, tji, xji, oji, t, fac, fl, f1, f2, beta, di, tlow, thigh
118 real(dp) :: b, g1, g2, vvjk, tjk, xjk
119 real(dp) :: vji(3), sji(3), alp(3), vb(3), vjk(3), sjk(3), vc(3)
120 real(dp) :: rho, ctheta, stheta, cphi, sphi
121 real(dp), external :: dnrm2
122
123 tlow = one - pt5*(one - params % se)*params % eta
124 thigh = one + pt5*(one + params % se)*params % eta
125
126 do ig = 1, params % ngrid
127 vb = zero
128 vc = zero
129 do ji = constants % inl(isph), constants % inl(isph+1) - 1
130 jsph = constants % nl(ji)
131 vji = params % csph(:,jsph) + &
132 & params % rsph(jsph)*constants % cgrid(:,ig) - &
133 & params % csph(:,isph)
134 !vvji = sqrt(dot_product(vji,vji))
135 vvji = dnrm2(3, vji, 1)
136 tji = vvji/params % rsph(isph)
137
138 if (tji.gt.thigh) cycle
139
140 sji = vji/vvji
141 !call dbasis(sji,basloc,dbsloc,vplm,vcos,vsin)
142 call dbasis(params, constants, sji, basloc, dbsloc, vplm, vcos, vsin)
143
144 alp = zero
145 t = one
146 do l = 1, params % lmax
147 ind = l*l + l + 1
148 fl = dble(l)
149 fac = t/(constants % vscales(ind)**2)
150 do m = -l, l
151 f2 = fac*sigma(ind+m,isph)
152 f1 = f2*fl*basloc(ind+m)
153 alp = alp + f1*sji + f2*dbsloc(:,ind+m)
154 end do
155 t = t*tji
156 end do
157 xji = fsw(tji, params % se, params % eta)
158 if (constants % fi(ig,jsph).gt.one) then
159 oji = xji/constants % fi(ig,jsph)
160 else
161 oji = xji
162 end if
163 f1 = oji/params % rsph(isph)
164 vb = vb + f1*alp*xi(ig,jsph)
165 if (tji .gt. tlow) then
166 beta = intmlp(params, constants, tji, sigma(:,isph), basloc)
167 if (constants % fi(ig,jsph) .gt. one) then
168 di = one/constants % fi(ig,jsph)
169 fac = di*xji
170 proc = .false.
171 b = zero
172 do jk = constants % inl(jsph), constants % inl(jsph+1) - 1
173 ksph = constants % nl(jk)
174 vjk = params % csph(:,jsph) + &
175 & params % rsph(jsph)*constants % cgrid(:,ig) - &
176 & params % csph(:,ksph)
177 !vvjk = sqrt(dot_product(vjk,vjk))
178 vvjk = dnrm2(3, vjk, 1)
179 tjk = vvjk/params % rsph(ksph)
180 if (ksph.ne.isph) then
181 if (tjk .le. thigh) then
182 proc = .true.
183 sjk = vjk/vvjk
184 !call ylmbas(sjk,basloc,vplm,vcos,vsin)
185 call ylmbas(sjk, rho, ctheta, stheta, cphi, sphi, &
186 & params % lmax, constants % vscales, basloc, vplm, &
187 & vcos, vsin)
188 g1 = intmlp(params, constants, tjk, sigma(:,ksph), basloc)
189 xjk = fsw(tjk, params % se, params % eta)
190 b = b + g1*xjk
191 end if
192 end if
193 end do
194 if (proc) then
195 g1 = di*di*dfsw(tji, params % se, params % eta)/params % rsph(isph)
196 g2 = g1*xi(ig,jsph)*b
197 vc = vc + g2*sji
198 end if
199 else
200 di = one
201 fac = zero
202 end if
203 f2 = (one-fac)*di*dfsw(tji, params % se, params % eta)/params % rsph(isph)
204 vb = vb + f2*xi(ig,jsph)*beta*sji
205 end if
206 end do
207 fx = fx + constants % wgrid(ig)*(vb - vc)
208 end do
209end subroutine contract_gradi_lji

345subroutine contract_gradi_bik(params, constants, isph, Xe, Xadj_e, force)
346 !! input/output
347 type(ddx_params_type), intent(in) :: params
348 type(ddx_constants_type), intent(in) :: constants
349 integer, intent(in) :: isph
350 real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: Xe
351 real(dp), dimension(params % ngrid), intent(in) :: Xadj_e
352 real(dp), dimension(3), intent(inout) :: force
353
354 ! Local Variables
355 integer :: igrid, ineigh, jsph
356 real(dp), dimension(0:params % lmax) :: SI_rijn
357 real(dp), dimension(0:params % lmax) :: DI_rijn
358 ! beta : Eq.(53) Stamm.etal.18
359 ! tlow : Lower bound for switch region
360 ! thigh : Upper bound for switch region
361 ! f1 : First factor in alpha computation
362 ! f2 : Second factor in alpha computation
363 real(dp) :: rijn, tij, beta, tlow, thigh, xij, oij, f1, f2, f3
364 ! alpha : Eq.(52) Stamm.etal.18
365 ! va : Eq.(54) Stamm.etal.18
366 real(dp) :: vij(3), sij(3), alpha(3), va(3), rj, vtij(3)
367 real(dp), external :: dnrm2
368 real(dp) :: work(params % lmax+1)
369 complex(dp) :: work_complex(params % lmax+1)
370
371 si_rijn = 0
372 di_rijn = 0
373
374 tlow = one - pt5*(one - params % se)*params % eta
375 thigh = one + pt5*(one + params % se)*params % eta
376
377 ! Loop over grid points
378 do igrid = 1, params % ngrid
379 va = zero
380 do ineigh = constants % inl(isph), constants % inl(isph+1) - 1
381 jsph = constants % nl(ineigh)
382 vij = params % csph(:,isph) + &
383 & params % rsph(isph)*constants % cgrid(:,igrid) - &
384 & params % csph(:,jsph)
385 rijn = dnrm2(3, vij, 1)
386 tij = rijn/params % rsph(jsph)
387 rj = params % rsph(jsph)
388 if (tij.ge.thigh) cycle
389 ! Computation of modified spherical Bessel function values
390 sij = vij/rijn
391 vtij = vij*params % kappa
392 call fmm_l2p_bessel_grad(vtij, params % rsph(jsph)*params % kappa, &
393 & params % lmax, constants % vscales, params % kappa, &
394 & xe(:, jsph), zero, alpha)
395 call fmm_l2p_bessel_work(vtij, params % lmax, constants % vscales, &
396 & constants % SI_ri(:, jsph), one, xe(:, jsph), zero, beta, &
397 & work_complex, work)
398 xij = fsw(tij, params % se, params % eta)
399 if (constants % fi(igrid,isph).gt.one) then
400 oij = xij/constants % fi(igrid,isph)
401 f2 = -oij/constants % fi(igrid,isph)
402 else
403 oij = xij
404 f2 = zero
405 end if
406 f1 = oij
407 va(:) = va(:) + f1*alpha(:) + beta*f2*constants % zi(:,igrid,isph)
408 if (tij .gt. tlow) then
409 f3 = beta*dfsw(tij,params % se,params % eta)/params % rsph(jsph)
410 if (constants % fi(igrid,isph).gt.one) &
411 & f3 = f3/constants % fi(igrid,isph)
412 va(:) = va(:) + f3*sij(:)
413 end if
414 end do
415 force = force - constants % wgrid(igrid)*xadj_e(igrid)*va(:)
416 end do
417end subroutine contract_gradi_bik

429subroutine contract_gradi_bji(params, constants, isph, Xe, Xadj_e, force)
430 !! input/output
431 type(ddx_params_type), intent(in) :: params
432 type(ddx_constants_type), intent(in) :: constants
433 integer, intent(in) :: isph
434 real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: Xe
435 real(dp), dimension(params % ngrid, params % nsph), intent(in) :: Xadj_e
436 real(dp), dimension(3), intent(inout) :: force
437
438 ! Local Variables
439 ! jk : Row pointer over kth row
440 integer :: igrid, jsph, ksph, ineigh, jk
441 real(dp), dimension(0:params % lmax) :: SI_rjin, SI_rjkn
442 real(dp), dimension(0:params % lmax) :: DI_rjin, DI_rjkn
443
444 logical :: proc
445 ! fac : \delta_fj_n*\omega^\eta_ji
446 ! f1 : First factor in alpha computation
447 ! f2 : Second factor in alpha computation
448 ! beta_ji : Eq.(57) Stamm.etal.18
449 ! dj : Before Eq.(10) Stamm.etal.18
450 real(dp) :: rjin, tji, xji, oji, fac, f1, f2, beta_ji, dj, tlow, thigh
451 real(dp) :: b, beta_jk, g1, g2, rjkn, tjk, xjk
452 ! alpha : Eq.(56) Stamm.etal.18
453 ! vb : Eq.(60) Stamm.etal.18
454 ! vc : Eq.(59) Stamm.etal.18
455 real(dp) :: vji(3), sji(3), vjk(3), alpha(3), vb(3), vc(3), &
456 & vtji(3), vtjk(3)
457 ! rho : Argument for ylmbas
458 ! ctheta : Argument for ylmbas
459 ! stheta : Argument for ylmbas
460 ! cphi : Argument for ylmbas
461 ! sphi : Argument for ylmbas
462 real(dp) :: ri
463
464 real(dp), external :: dnrm2
465 real(dp) :: work(params % lmax+1)
466 complex(dp) :: work_complex(params % lmax+1)
467
468 si_rjin = 0
469 di_rjin = 0
470 si_rjkn = 0
471 di_rjkn = 0
472
473 tlow = one - pt5*(one - params % se)*params % eta
474 thigh = one + pt5*(one + params % se)*params % eta
475
476 do igrid = 1, params % ngrid
477 vb = zero
478 vc = zero
479 do ineigh = constants % inl(isph), constants % inl(isph+1) - 1
480 jsph = constants % nl(ineigh)
481 vji = params % csph(:,jsph) + &
482 & params % rsph(jsph)*constants % cgrid(:,igrid) - &
483 & params % csph(:,isph)
484 rjin = dnrm2(3, vji, 1)
485 ri = params % rsph(isph)
486 tji = rjin/ri
487
488 if (tji.gt.thigh) cycle
489
490 sji = vji/rjin
491 vtji = vji*params % kappa
492 call fmm_l2p_bessel_grad(vtji, params % rsph(isph)*params % kappa, &
493 & params % lmax, constants % vscales, params % kappa, &
494 & xe(:, isph), zero, alpha)
495 xji = fsw(tji,params % se,params % eta)
496 if (constants % fi(igrid,jsph).gt.one) then
497 oji = xji/constants % fi(igrid,jsph)
498 else
499 oji = xji
500 end if
501 f1 = oji
502 vb = vb + f1*alpha*xadj_e(igrid,jsph)
503 if (tji .gt. tlow) then
504 call fmm_l2p_bessel_work(vtji, params % lmax, &
505 & constants % vscales, constants % SI_ri(:, isph), one, &
506 & xe(:, isph), zero, beta_ji, work_complex, work)
507 if (constants % fi(igrid,jsph) .gt. one) then
508 dj = one/constants % fi(igrid,jsph)
509 fac = dj*xji
510 proc = .false.
511 b = zero
512 do jk = constants % inl(jsph), constants % inl(jsph+1) - 1
513 ksph = constants % nl(jk)
514 vjk = params % csph(:,jsph) + &
515 & params % rsph(jsph)*constants % cgrid(:,igrid) - &
516 & params % csph(:,ksph)
517 rjkn = dnrm2(3, vjk, 1)
518 tjk = rjkn/params % rsph(ksph)
519 if (ksph.ne.isph) then
520 if (tjk .le. thigh) then
521 proc = .true.
522 vtjk = vjk*params % kappa
523 call fmm_l2p_bessel_work(vtjk, params % lmax, &
524 & constants % vscales, &
525 & constants % SI_ri(:, ksph), one, xe(:, ksph), &
526 & zero, beta_jk, work_complex, work)
527 xjk = fsw(tjk, params % se, params % eta)
528 b = b + beta_jk*xjk
529 end if
530 end if
531 end do
532 if (proc) then
533 g1 = dj*dj*dfsw(tji,params % se,params % eta) &
534 & /params % rsph(isph)
535 g2 = g1*xadj_e(igrid,jsph)*b
536 vc = vc + g2*sji
537 end if
538 else
539 dj = one
540 fac = zero
541 end if
542 f2 = (one-fac)*dj*dfsw(tji,params % se,params % eta) &
543 & /params % rsph(isph)
544 vb = vb + f2*xadj_e(igrid,jsph)*beta_ji*sji
545 end if
546 end do
547 force = force + constants % wgrid(igrid)*(vb - vc)
548 end do
549end subroutine contract_gradi_bji

562subroutine contract_grad_c_worker1(params, constants, workspace, Xadj_r_sgrid, &
563 & Xadj_e_sgrid, diff_re, force)
564 !! Inputs
565 type(ddx_params_type), intent(in) :: params
566 type(ddx_constants_type), intent(in) :: constants
567 type(ddx_workspace_type), intent(inout) :: workspace
568 real(dp), dimension(constants % nbasis, params % nsph), intent(in) :: &

571 & Xadj_r_sgrid, Xadj_e_sgrid

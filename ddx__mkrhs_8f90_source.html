23 type(ddx_constants_type), intent(in) :: constants
24 integer, intent(in) :: phi_flag, grad_flag, hessian_flag
25 real(dp), intent(out) :: phi_cav(constants % ncav)
26 real(dp), intent(out) :: gradphi_cav(3, constants % ncav)
27 real(dp), intent(out) :: hessianphi_cav(3, 3, constants % ncav)
28 real(dp), intent(out) :: psi(constants % nbasis, &
29 & params % nsph)
30 ! Local variables
31 integer :: isph, igrid, icav, inode, inear, jnear, jnode, jsph, i
32 real(dp) :: d(3), v, tmpv, r, gradv(3), hessianv(3, 3), tmpd(3), epsp=one
33 real(dp), allocatable :: grid_grad(:,:,:), grid_hessian(:,:,:,:), &
34 & grid_hessian2(:,:,:)
35 real(dp), external :: dnrm2
36 real(dp) :: t

38 if (grad_flag .eq. 1) allocate(grid_grad(params % ngrid, 3, &
39 & params % nsph))
40 if (hessian_flag .eq. 1) allocate(grid_hessian(params % ngrid, &
41 & 3, 3, params % nsph), grid_hessian2(params % ngrid, &
42 & 3, params % nsph))

44 ! In case FMM is disabled compute phi and gradphi at cavity points by a
45 ! naive double loop of a quadratic complexity
46 if (params % fmm .eq. 0) then
47 do icav = 1, constants % ncav
48 v = zero
49 gradv = zero
50 hessianv = zero
51 do isph = 1, params % nsph
52 d = constants % ccav(:, icav) - &
53 & params % csph(:, isph)
54 r = dnrm2(3, d, 1)
55 d = d / r
56 tmpv = params % charge(isph) / r
57 v = v + tmpv
58 tmpv = tmpv / r
59 tmpd = tmpv * d
60 tmpv = tmpv / r
61 gradv = gradv - tmpd
62 tmpd = three / r * tmpd
63 hessianv(:, 1) = hessianv(:, 1) + tmpd*d(1)
64 hessianv(:, 2) = hessianv(:, 2) + tmpd*d(2)
65 hessianv(:, 3) = hessianv(:, 3) + tmpd*d(3)
66 hessianv(1, 1) = hessianv(1, 1) - tmpv
67 hessianv(2, 2) = hessianv(2, 2) - tmpv
68 hessianv(3, 3) = hessianv(3, 3) - tmpv
69 end do
70 if (phi_flag .eq. 1) then
71 phi_cav(icav) = v
72 end if
73 if (grad_flag .eq. 1) then
74 gradphi_cav(:, icav) = gradv
75 end if
76 if (hessian_flag .eq. 1) then
77 hessianphi_cav(:, :, icav) = hessianv
78 end if
79 end do
80 ! Use the FMM otherwise
81 else
82 ! P2M step from centers of harmonics
83 do isph = 1, params % nsph
84 inode = constants % snode(isph)
85 workspace % tmp_sph(1, isph) = &
86 & params % charge(isph) &
87 & / params % rsph(isph) / sqrt4pi
88 workspace % tmp_sph(2:, isph) = zero
89 workspace % tmp_node_m(1, inode) = &
90 & workspace % tmp_sph(1, isph)
91 workspace % tmp_node_m(2:, inode) = zero
92 end do
93 ! M2M, M2L and L2L translations
94
95 call tree_m2m_rotation(params, constants, &
96 & workspace % tmp_node_m)
97
98 call tree_m2l_rotation(params, constants, &
99 & workspace % tmp_node_m, workspace % tmp_node_l)
100
101 call tree_l2l_rotation(params, constants, &
102 & workspace % tmp_node_l)
103
104 call tree_l2p(params, constants, one, &
105 & workspace % tmp_node_l, zero, &
106 & workspace % tmp_grid, workspace % tmp_sph_l)
107
108 call tree_m2p(params, constants, &
109 & params % lmax, one, workspace % tmp_sph, &
110 & one, workspace % tmp_grid)
111
112 ! Potential from each sphere to its own grid points
113 call dgemm('T', 'N', params % ngrid, params % nsph, &
114 & constants % nbasis, one, constants % vgrid2, &
115 & constants % vgrid_nbasis, workspace % tmp_sph, &
116 & constants % nbasis, one, workspace % tmp_grid, &
117 & params % ngrid)
118
119 ! Rearrange potential from all grid points to external only
120 if (phi_flag.eq.1) then
121 icav = 0
122 do isph = 1, params % nsph
123 do igrid = 1, params % ngrid
124 ! Do not count internal grid points
125 if(constants % ui(igrid, isph) .eq. zero) cycle
126 icav = icav + 1
127 phi_cav(icav) = workspace % tmp_grid(igrid, isph)
128 end do
129 end do
130 end if
131
132 ! Now compute near-field FMM gradients and hessians
133 ! Cycle over all spheres
134 if (grad_flag.eq.1 .or. hessian_flag.eq.1) then
135 if (grad_flag.eq.1) grid_grad(:, :, :) = zero
136 if (hessian_flag.eq.1) grid_hessian(:, :, :, :) = zero
137 !$omp parallel do default(none) shared(params,constants,workspace, &
138 !$omp grid_grad,grid_hessian,grad_flag,hessian_flag) &
139 !$omp private(isph,igrid,inode,jnear,jnode,jsph,d,r,tmpd,tmpv) &
140 !$omp schedule(dynamic)
141 do isph = 1, params % nsph
142 ! Cycle over all external grid points
143 do igrid = 1, params % ngrid
144 if(constants % ui(igrid, isph) .eq. zero) cycle
145 ! Cycle over all near-field admissible pairs of spheres,
146 ! including pair (isph, isph) which is a self-interaction
147 inode = constants % snode(isph)
148 do jnear = constants % snear(inode), &
149 & constants % snear(inode+1) - 1
150 ! Near-field interactions are possible only between leaf
151 ! nodes, which must contain only a single input sphere
152 jnode = constants % near(jnear)
153 jsph = constants % order( &
154 & constants % cluster(1, jnode))
155 d = params % csph(:, isph) + &
156 & constants % cgrid(:, igrid) &
157 & *params % rsph(isph) &
158 & - params % csph(:, jsph)
159! r = dnrm2(3, d, 1)
160 r = sqrt(d(1)*d(1) + d(2)*d(2) + d(3)*d(3))
161 d = d / r / r
162 tmpv = params % charge(jsph) / r
163 if (grad_flag.eq.1) grid_grad(igrid, :, isph) = &
164 & grid_grad(igrid, :, isph) - tmpv * d
165 tmpd = three * tmpv * d
166 tmpv = tmpv / r / r
167 if (hessian_flag.eq.1) then
168 grid_hessian(igrid, 1, :, isph) = &
169 & grid_hessian(igrid, 1, :, isph) + d(1)*tmpd
170 grid_hessian(igrid, 2, :, isph) = &
171 & grid_hessian(igrid, 2, :, isph) + d(2)*tmpd
172 grid_hessian(igrid, 3, :, isph) = &
173 & grid_hessian(igrid, 3, :, isph) + d(3)*tmpd
174 grid_hessian(igrid, 1, 1, isph) = &
175 & grid_hessian(igrid, 1, 1, isph) - tmpv
176 grid_hessian(igrid, 2, 2, isph) = &
177 & grid_hessian(igrid, 2, 2, isph) - tmpv
178 grid_hessian(igrid, 3, 3, isph) = &
179 & grid_hessian(igrid, 3, 3, isph) - tmpv
180 end if
181 end do
182 end do
183 end do
184 end if
185
186 ! Take into account far-field FMM gradients only if pl > 0
187 if ((params % pl .gt. 0) .and. (grad_flag.eq.1)) then
188 ! Get gradient of L2L
189 call tree_grad_l2l(params, constants, &
190 & workspace % tmp_node_l, &
191 & workspace % tmp_sph_l_grad, &
192 & workspace % tmp_sph_l)
193 ! Apply L2P for every axis with -1 multiplier since grad over
194 ! target point is equal to grad over source point
195 call dgemm('T', 'N', params % ngrid, &
196 & 3*params % nsph, (params % pl)**2, &
197 & -one, constants % vgrid2, &
198 & constants % vgrid_nbasis, &
199 & workspace % tmp_sph_l_grad, &
200 & (params % pl+1)**2, one, grid_grad, &
201 & params % ngrid)
202 end if
203 ! Take into account far-field FMM hessians only if pl > 1
204 if ((params % pl .gt. 1) .and. (hessian_flag.eq.1)) then
205 do i = 1, 3
206 ! Load previously computed gradient into leaves, since
207 ! tree_grad_l2l currently takes local expansions of entire
208 ! tree. In future it might be changed.
209 do isph = 1, params % nsph
210 inode = constants % snode(isph)
211 workspace % tmp_node_l(:, inode) = &
212 & workspace % tmp_sph_l_grad(:, i, isph)
213 end do
214 ! Get gradient of a gradient of L2L. Currently this uses input
215 ! pl maximal degree of local harmonics but in reality we need
216 ! only pl-1 maximal degree since coefficients of harmonics of a
217 ! degree pl are zeros.
218 call tree_grad_l2l(params, constants, &
219 & workspace % tmp_node_l, &
220 & workspace % tmp_sph_l_grad2, &
221 & workspace % tmp_sph_l)
222 ! Apply L2P for every axis
223 call dgemm('T', 'N', params % ngrid, &
224 & 3*params % nsph, (params % pl-1)**2, &
225 & one, constants % vgrid2, &
226 & constants % vgrid_nbasis, &
227 & workspace % tmp_sph_l_grad2, &
228 & (params % pl+1)**2, zero, grid_hessian2, &
229 & params % ngrid)
230 ! Properly copy hessian
231 grid_hessian(:, i, :, :) = grid_hessian(:, i, :, :) + grid_hessian2
232 end do
233 end if
234
235 ! Copy output for external grid points only
236 icav = 0
237 do isph = 1, params % nsph
238 do igrid = 1, params % ngrid
239 if(constants % ui(igrid, isph) .eq. zero) cycle
240 icav = icav + 1
241 if (grad_flag .eq. 1) gradphi_cav(:, icav) = &
242 & grid_grad(igrid, :, isph)
243 if (hessian_flag .eq. 1) hessianphi_cav(:, :, icav) = &
244 & grid_hessian(igrid, :, :, isph)
245 end do
246 end do
247 end if
248
249 ! Vector psi
250 psi(2:, :) = zero
251 do isph = 1, params % nsph
252 psi(1, isph) = sqrt4pi * params % charge(isph)
253 end do
254
255 ! deallocate temporary
256 if (grad_flag .eq. 1) deallocate(grid_grad)
257 if (hessian_flag .eq. 1) deallocate(grid_hessian, grid_hessian2)
258end subroutine mkrhs
